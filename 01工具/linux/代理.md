可以采用`xray`
设置`/usr/local/etc/xray/config.json`
```
{
  "log": {
    "loglevel": "warning"
  },
  "inbounds": [
    {
      "listen": "127.0.0.1",
      "port": 10809,
      "protocol": "socks",
      "settings": {
        "udp": true
      }
    },
    {
      "listen": "127.0.0.1",
      "port": 10810,
      "protocol": "http"
    }
  ],
  "outbounds": [
    {
      "protocol": "vmess",
      "settings": {
        "vnext": [
          {
            "address": "****需要在v2rayN的节点编辑中找域名****",
            "port": ****需要在v2rayN的节点编辑中找端口****,
            "users": [
              {
                "id": "****需要在v2rayN的节点编辑中找用户****",
                "alterId": 0,
                "security": "auto"
              }
            ]
          }
        ]
      },
      "streamSettings": {
        "network": "tcp",
        "tcpSettings": {
          "header": {
            "type": "http",
            "request": {
              "path": ["/"],
              "headers": {
                "Host": ["www.cloudflare.com"]
              }
            }
          }
        }
      }
    }
  ]
}
```
### xray是否活动
```bash
sudo systemctl status xray
```

### 启动它
```bash
sudo systemctl start xray
```
