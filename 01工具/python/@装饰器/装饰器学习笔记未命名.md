## 1) 函数是一等公民 & 闭包（基础原理）
```python
def outer(x):
    def inner(y):
        return x + y
    return inner

add_five = outer(5) # 返回的可以视为inner函数了
print(add_five(3))
```
**示例输出**
```
8
```
**分析**：`inner` 记住了 `x=5`（闭包）。装饰器本质：接收函数，返回新函数，常用闭包保存状态。

## 2) 无参数装饰器模板（最常用）
```python
import functools

def log(func):
	@functools.wraps(func) # functools.wraps 会把被装饰函数的关键元数据复制到 wrapper 上
	def wrapper(*args, **kwargs):
		print(f"[log] calling {func.__name__}")
		result = func(*args, **kwargs)
		print(f"[log] {func.__name__} done")
		return result
	return wrapper  # 关键，这里是调用了 wrapper 函数了

@log  # 等价于 hello = log(hello)
def hello(name):
	print(f"Hello, {name}!")

hello("Alice")
```
**示例输出**
```
[log] calling hello
Hello, Alice!
[log] hello done
```
**分析**：`@wraps` 保留原函数元数据；`*args, **kwargs` 兼容任意签名；`wrapper` 返回原函数结果。

## 3) 带参数装饰器模板（三层结构）
```python
import functools

def repeat(times: int):
    def decorator(func):
        @functools.wraps(func)
        def wrapper(*args, **kwargs):
            result = None
            for _ in range(times):
                result = func(*args, **kwargs)
            return result
        return wrapper
    return decorator

@repeat(3)
def greet(name):
    print(f"Hi, {name}!")

greet("Bob")
```
**示例输出**
```
Hi, Bob!
Hi, Bob!
Hi, Bob!
```
**分析**：外层 `repeat` 接收参数；中层拿到 `func`；内层 `wrapper` 执行。带参数的装饰器比无参多一层。

## 4) 叠加装饰器顺序
```python
def A(func):
    def wrap(*a, **k):
        print("A before")
        r = func(*a, **k)
        print("A after")
        return r
    return wrap

def B(func):
    def wrap(*a, **k):
        print("B before")
        r = func(*a, **k)
        print("B after")
        return r
    return wrap

@A
@B
def run():
    print("run")

run()
```
**示例输出**
```
A before
B before
run
B after
A after
```
**分析**：叠加时自下而上应用：`run = A(B(run))`。执行时外层先进入，内层后进入。

## 5) 实战案例（常用套路）

### 5.1 计时
```python
import time
import functools

def timed(func):
    @functools.wraps(func)
    def wrapper(*args, **kwargs):
        start = time.perf_counter()
        try:
            return func(*args, **kwargs)
        finally:
            dt = (time.perf_counter() - start) * 1000
            print(f"{func.__name__} took {dt:.2f} ms")
    return wrapper

@timed
def work(n):
    return sum(range(n))

work(1_000_000)
```
**示例输出**
```
work took 55.00 ms   # 具体时间因机器而异
```
**分析**：`try/finally` 确保即便异常也会打印耗时。

### 5.2 重试（带参数）
```python
import functools
import time

def retry(times=3, delay=0.1):
    def decorator(func):
        @functools.wraps(func)
        def wrapper(*args, **kwargs):
            last_exc = None
            for attempt in range(1, times + 1):
                try:
                    return func(*args, **kwargs)
                except Exception as exc:
                    last_exc = exc
                    print(f"attempt {attempt} failed: {exc}")
                    time.sleep(delay)
            raise last_exc
        return wrapper
    return decorator

@retry(times=2, delay=0.2)
def might_fail(x):
    if x < 0:
        raise ValueError("x must be non-negative")
    return x

might_fail(-1)
```
**示例输出**
```
attempt 1 failed: x must be non-negative
attempt 2 failed: x must be non-negative
Traceback (most recent call last):
  ...
ValueError: x must be non-negative
```
**分析**：失败会按次数重试，最终抛出最后一次异常；`delay` 控制间隔。

### 5.3 缓存：`functools.lru_cache`
```python
import functools

@functools.lru_cache(maxsize=128)
def fib(n: int) -> int:
    if n < 2:
        return n
    return fib(n - 1) + fib(n - 2)

print(fib(10))
print(fib.cache_info())
```
**示例输出**
```
55
CacheInfo(hits=8, misses=11, maxsize=128, currsize=11)
```
**分析**：重复子问题被缓存；`cache_info` 可查看命中率。

### 5.4 运行前校验
```python
import functools

def require_positive(func):
    @functools.wraps(func)
    def wrapper(x: int):
        if x <= 0:
            raise ValueError("x must be positive")
        return func(x)
    return wrapper

@require_positive
def sqrt(x):
    return x ** 0.5

print(sqrt(9))
sqrt(-1)
```
**示例输出**
```
3.0
Traceback (most recent call last):
  ...
ValueError: x must be positive
```
**分析**：在进入核心逻辑前做参数检查，违规直接抛出异常。

### 5.5 简单鉴权（示意）
```python
import functools

def require_role(role):
    def decorator(func):
        @functools.wraps(func)
        def wrapper(user, *args, **kwargs):
            if role not in user.get("roles", []):
                raise PermissionError(f"need role: {role}")
            return func(user, *args, **kwargs)
        return wrapper
    return decorator

@require_role("admin")
def delete_user(user, target_id):
    print(f"delete {target_id}")

user = {"name": "Tom", "roles": ["admin"]}
delete_user(user, 123)
```
**示例输出**
```
delete 123
```
**分析**：通过闭包捕获所需角色，运行时校验用户权限。

## 6) 面向类的装饰器

### 6.1 装饰实例方法
```python
import functools

def log_method(func):
    @functools.wraps(func)
    def wrapper(self, *args, **kwargs):
        print(f"[{self}] -> {func.__name__}")
        return func(self, *args, **kwargs)
    return wrapper

class Greeter:
    @log_method
    def hello(self, name):
        print(f"hello {name}")

Greeter().hello("Coco")
```
**示例输出**
```
[<__main__.Greeter object at 0x...>] -> hello
hello Coco
```
**分析**：记得在 `wrapper` 的第一个参数接收 `self`。

### 6.2 类装饰器（修改类）
```python
def add_repr(cls):
    def __repr__(self):
        fields = ", ".join(f"{k}={v!r}" for k, v in self.__dict__.items())
        return f"{cls.__name__}({fields})"
    cls.__repr__ = __repr__
    return cls

@add_repr
class Point:
    def __init__(self, x, y):
        self.x, self.y = x, y

print(Point(1, 2))
```
**示例输出**
```
Point(x=1, y=2)
```
**分析**：类装饰器直接接收类，可增删改类的属性或方法。

### 6.3 用 `__call__` 让类实例当装饰器（可携带状态）
```python
import functools
import time

class Timer:
    def __init__(self, unit="ms"):
        self.unit = unit
        self.scale = 1000 if unit == "ms" else 1

    def __call__(self, func):
        @functools.wraps(func)
        def wrapper(*args, **kwargs):
            start = time.perf_counter()
            try:
                return func(*args, **kwargs)
            finally:
                dt = (time.perf_counter() - start) * self.scale
                print(f"{func.__name__} took {dt:.2f} {self.unit}")
        return wrapper

@Timer(unit="s")
def slow():
    time.sleep(0.2)

slow()
```
**示例输出**
```
slow took 0.20 s
```
**分析**：状态（如单位、配置）存在实例属性里，便于复用。

## 7) 进阶：同步/异步兼容、可组合

### 7.1 同步 + 异步兼容
```python
import functools
import inspect
import asyncio

def log_call(func):
    @functools.wraps(func)
    async def async_wrapper(*args, **kwargs):
        print(f"async {func.__name__}")
        return await func(*args, **kwargs)

    @functools.wraps(func)
    def sync_wrapper(*args, **kwargs):
        print(f"sync {func.__name__}")
        return func(*args, **kwargs)

    return async_wrapper if inspect.iscoroutinefunction(func) else sync_wrapper

@log_call
async def fetch():
    await asyncio.sleep(0.1)
    return "data"

@log_call
def compute():
    return 42

async def main():
    print(await fetch())
    print(compute())

asyncio.run(main())
```
**示例输出（异步需事件循环）**
```
async fetch
data
sync compute
42
```
**分析**：用 `inspect.iscoroutinefunction` 判断是否异步，返回不同的 wrapper。

### 7.2 组合多个装饰器生成一个
```python
def compose(*decorators):
    def combined(func):
        for deco in reversed(decorators):  # 反向应用，保持直观顺序
            func = deco(func)
        return func
    return combined

@compose(timed, retry(times=2))  # timed / retry 复用前文定义
def risky():
    raise RuntimeError("boom")

risky()
```
**示例输出**
```
attempt 1 failed: boom
attempt 2 failed: boom
risky took 0.40 ms
Traceback (most recent call last):
  ...
RuntimeError: boom
```
**分析**：`compose` 把多个装饰器打包成一个，便于复用固定的装饰组合。

## 8) 常用内置装饰器

### 8.1 `@property`
```python
class Circle:
    def __init__(self, r):
        self.r = r

    @property
    def area(self):
        return 3.14159 * self.r * self.r

c = Circle(2)
print(c.area)
```
**示例输出**
```
12.56636
```
**分析**：方法改为只读属性，访问时自动计算。

### 8.2 `@functools.lru_cache`（已示于 5.3）

### 8.3 `@functools.singledispatch`（基于类型的函数重载）
```python
from functools import singledispatch

@singledispatch
def dump(value):
    raise TypeError("unsupported")

@dump.register(int)
def _(value: int):
    print(f"int {value}")

@dump.register(list)
def _(value: list):
    print(f"list of {len(value)} items")

dump(3)
dump([1, 2, 3])
```
**示例输出**
```
int 3
list of 3 items
```
**分析**：根据参数类型自动分派到不同实现，便于扩展。

### 8.4 其它常见
- `@staticmethod` / `@classmethod`：静态方法、类方法。
- `@functools.cached_property`：属性首次计算后缓存。
- `@functools.partial`（不是装饰器但常配合）：预填部分参数。

## 9) 调试与常见坑
- 忘记 `@wraps`：丢失 `__name__`/`__doc__`，影响文档、调试、类型提示。
- `wrapper` 未返回原函数结果：导致调用方拿到 `None`。
- 参数不通用：缺少 `*args, **kwargs` 会限制使用范围。
- 叠加顺序搞错：记住装饰是自下而上绑定，执行是外层先进入。
- 异步函数用同步 wrapper：会返回未 await 的协程对象。
- 共享可变状态要注意线程/协程安全（锁或无共享）。
