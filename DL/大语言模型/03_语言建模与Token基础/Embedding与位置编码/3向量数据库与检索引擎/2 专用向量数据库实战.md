## 五大主流产品

| 产品 | 特点 | 适合场景 |
| --- | --- | --- |
| **Pinecone** | 全托管、生产就绪、集成 hybrid/rerank | 快速上线、不想运维 |
| **Qdrant** | 开源、HNSW+payload filter、named vectors | 灵活自部署、多租户 |
| **Milvus** | 开源、大规模、索引类型全 | 十亿级数据、工程生态 |
| **Weaviate** | hybrid、多模态、schema 体验好 | 多模态、语义搜索 |
| **Chroma** | 本地轻量、开发简单 | 原型开发、个人项目 |

## Qdrant 实战

```python
from qdrant_client import QdrantClient
from qdrant_client.models import VectorParams, Distance, PointStruct
import numpy as np

# 连接（内存模式，适合开发）
client = QdrantClient(":memory:")

# 创建 collection
client.create_collection(
    collection_name="my_docs",
    vectors_config=VectorParams(
        size=768,
        distance=Distance.COSINE
    )
)

# 插入向量（带 metadata/payload）
points = [
    PointStruct(
        id=i,
        vector=np.random.randn(768).tolist(),
        payload={"title": f"Document {i}", "category": "tech", "year": 2024}
    )
    for i in range(100)
]
client.upsert(collection_name="my_docs", points=points)

# 搜索（带 metadata filter）
from qdrant_client.models import Filter, FieldCondition, MatchValue

results = client.search(
    collection_name="my_docs",
    query_vector=np.random.randn(768).tolist(),
    query_filter=Filter(
        must=[
            FieldCondition(key="category", match=MatchValue(value="tech")),
        ]
    ),
    limit=5
)

for r in results:
    print(f"ID: {r.id}, Score: {r.score:.4f}, Title: {r.payload['title']}")
```

## Pinecone 实战

```python
from pinecone import Pinecone, ServerlessSpec
import numpy as np

# 初始化
pc = Pinecone(api_key="YOUR_API_KEY")

# 创建 index
pc.create_index(
    name="my-index",
    dimension=768,
    metric="cosine",
    spec=ServerlessSpec(cloud="aws", region="us-east-1")
)
index = pc.Index("my-index")

# 插入
vectors = [
    {
        "id": f"doc_{i}",
        "values": np.random.randn(768).tolist(),
        "metadata": {"category": "tech", "year": 2024}
    }
    for i in range(100)
]
index.upsert(vectors=vectors)

# 搜索（带 metadata filter）
results = index.query(
    vector=np.random.randn(768).tolist(),
    top_k=5,
    filter={"category": {"$eq": "tech"}},
    include_metadata=True
)

for match in results["matches"]:
    print(f"ID: {match['id']}, Score: {match['score']:.4f}")
```

## Milvus 实战

```python
from pymilvus import MilvusClient
import numpy as np

# 连接（Milvus Lite 本地模式）
client = MilvusClient("milvus_demo.db")

# 创建 collection
client.create_collection(
    collection_name="my_docs",
    dimension=768,
    metric_type="COSINE"
)

# 插入数据
data = [
    {
        "id": i,
        "vector": np.random.randn(768).tolist(),
        "title": f"Document {i}",
        "category": "tech"
    }
    for i in range(100)
]
client.insert(collection_name="my_docs", data=data)

# 搜索
results = client.search(
    collection_name="my_docs",
    data=[np.random.randn(768).tolist()],
    limit=5,
    output_fields=["title", "category"]
)

for hits in results:
    for hit in hits:
        print(f"ID: {hit['id']}, Distance: {hit['distance']:.4f}")
```

## Chroma 实战

```python
import chromadb

# 本地内存客户端
client = chromadb.Client()

# 创建 collection
collection = client.create_collection(
    name="my_docs",
    metadata={"hnsw:space": "cosine"}
)

# 插入（Chroma 可自动调用 embedding 函数）
collection.add(
    ids=[f"doc_{i}" for i in range(10)],
    documents=[
        "Machine learning is a branch of AI",
        "Python is a popular programming language",
        "Deep learning uses neural networks",
        "Vector databases store embeddings",
        "Natural language processing handles text",
        "Computer vision processes images",
        "Reinforcement learning learns from rewards",
        "Transfer learning reuses pretrained models",
        "Transformers revolutionized NLP",
        "GPT is a large language model",
    ],
    metadatas=[{"category": "AI"} for _ in range(10)]
)

# 搜索
results = collection.query(
    query_texts=["What is artificial intelligence?"],
    n_results=3,
    where={"category": "AI"}
)

for doc, dist in zip(results["documents"][0], results["distances"][0]):
    print(f"Distance: {dist:.4f} | {doc[:60]}")
```
