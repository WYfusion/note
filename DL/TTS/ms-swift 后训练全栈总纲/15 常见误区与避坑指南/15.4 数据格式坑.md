## 2. 通用 schema 校验

```python
import json
from jsonschema import Draft202012Validator

SFT_SCHEMA = {
    'type': 'object',
    'required': ['messages'],
    'properties': {
        'messages': {
            'type': 'array',
            'minItems': 2,
            'items': {
                'type': 'object',
                'required': ['role', 'content'],
                'properties': {
                    'role': {'enum': ['system', 'user', 'assistant', 'tool']},
                    'content': {'type': ['string', 'null']},
                    'tool_calls': {'type': 'array'},
                },
            },
        },
    },
}

def validate(jsonl):
    v = Draft202012Validator(SFT_SCHEMA)
    bad = 0
    for ln, line in enumerate(open(jsonl), 1):
        try:
            r = json.loads(line)
        except json.JSONDecodeError as e:
            print(f'L{ln} JSON: {e}'); bad += 1; continue
        errs = list(v.iter_errors(r))
        if errs:
            print(f'L{ln} schema: {errs[0].message}')
            bad += 1
    print(f'bad={bad}')
validate('train.jsonl')
```

## 3. 多模态占位符检查

```python
import json

def check_mm(jsonl, modality='images'):
    placeholder = '<image>' if modality == 'images' else '<audio>' if modality == 'audios' else '<video>'
    bad = 0
    for ln, line in enumerate(open(jsonl), 1):
        r = json.loads(line)
        files = r.get(modality, [])
        n_token = sum((m.get('content') or '').count(placeholder) for m in r['messages'])
        if n_token != len(files):
            print(f'L{ln} 占位符 {n_token} 个，文件 {len(files)} 个')
            bad += 1
    print(f'bad={bad}')
check_mm('mm_train.jsonl', 'images')
```

## 4. 长度统计

```python
import json, statistics
from transformers import AutoTokenizer

tok = AutoTokenizer.from_pretrained('Qwen/Qwen2.5-7B-Instruct')
lens = []
for line in open('train.jsonl'):
    r = json.loads(line)
    text = '\n'.join((m.get('content') or '') for m in r['messages'])
    lens.append(len(tok.encode(text)))
lens.sort()
print(f'count: {len(lens)}')
print(f'p50: {lens[len(lens)//2]}')
print(f'p95: {lens[int(0.95*len(lens))]}')
print(f'p99: {lens[int(0.99*len(lens))]}')
print(f'max: {lens[-1]}')
# 选 max_length 时通常取 p95 + 一点 buffer
```

## 5. 训练 / 评测污染检测

```python
import hashlib, json

def sigs(jsonl, key_fn):
    s = set()
    for line in open(jsonl):
        r = json.loads(line)
        s.add(hashlib.sha1(key_fn(r).encode()).hexdigest()[:16])
    return s

def key_fn(r):
    # 用 user 提问归一化作为指纹
    user = next(m for m in r['messages'] if m['role'] == 'user')['content']
    return ''.join(user.split())

train_sigs = sigs('train.jsonl', key_fn)
eval_sigs = sigs('eval.jsonl', key_fn)
overlap = train_sigs & eval_sigs
print(f'overlap = {len(overlap)} / eval = {len(eval_sigs)}')
```

## 6. 偏好数据反向检查

```python
import json

def quick_check_pref(jsonl, n=20):
    rows = [json.loads(l) for l in open(jsonl)]
    import random
    for r in random.sample(rows, n):
        print('--- prompt ---')
        print(r['messages'][-1]['content'][:200])
        print('--- chosen   ---')
        print(r['chosen'][:200])
        print('--- rejected ---')
        print(r['rejected'][:200])
        print()
quick_check_pref('dpo.jsonl')
```
