## 5. 实验跟踪表

```python
import json, os, subprocess, hashlib
from datetime import datetime

def log_experiment(cfg_path: str, out_dir: str):
    cfg = open(cfg_path).read()
    code_sha = subprocess.check_output(['git', 'rev-parse', 'HEAD']).decode().strip()
    data_path = json.loads(open(cfg_path.replace('.yaml', '.json')).read()).get('dataset', '')
    data_sha = hashlib.sha1(open(data_path, 'rb').read()).hexdigest()[:8] if os.path.exists(data_path) else 'unknown'
    record = {
        'name': os.path.basename(out_dir),
        'started_at': datetime.now().isoformat(),
        'code_sha': code_sha,
        'data_sha': data_sha,
        'config': cfg,
    }
    with open(os.path.join(out_dir, 'experiment.json'), 'w') as f:
        json.dump(record, f, indent=2, ensure_ascii=False)
```

## 6. 模型注册中心

```python
import json, shutil, time
from pathlib import Path

REG = Path('registry')

def register_model(ckpt_dir: str, name: str, eval_summary: dict, tags: list):
    REG.mkdir(exist_ok=True)
    target = REG / name
    if target.exists():
        raise ValueError(f'{name} 已存在')
    shutil.copytree(ckpt_dir, target / 'model')
    meta = {
        'name': name,
        'created_at': time.strftime('%Y-%m-%d %H:%M:%S'),
        'eval': eval_summary,
        'tags': tags,
    }
    json.dump(meta, open(target / 'meta.json', 'w'), indent=2, ensure_ascii=False)

def list_models():
    rows = []
    for d in REG.iterdir():
        if (d / 'meta.json').exists():
            rows.append(json.load(open(d / 'meta.json')))
    return sorted(rows, key=lambda r: r['created_at'], reverse=True)
```

## 7. 实验对比仪表盘

```python
import pandas as pd, json, glob

rows = []
for f in glob.glob('out/*/eval/summary.json'):
    s = json.load(open(f))
    name = f.split('/')[1]
    row = {'experiment': name}
    for k, v in s['results'].items():
        row[k] = v['score']
    rows.append(row)
df = pd.DataFrame(rows).sort_values('experiment')
print(df.to_string(index=False))
```

> [!important]
> 
> - **不要在共享盘乱写**：所有实验产物用统一前缀（如用户名 / 项目名）。
> 
> - **配置同步 git**：YAML 入仓，超参变化即 commit。
> 
> - **日志保留至少 30 天**：方便回溯出错原因。
> 
> - **eval 摘要要落地**：训练结束自动跑 swift eval 写到产物目录，避免后期补不齐。
