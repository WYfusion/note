## 2. Step 1：Agent SFT

```bash
swift sft \
    --model Qwen/Qwen2.5-7B-Instruct \
    --dataset agentflan my-domain-agent \
    --dataset_ratio 1.0 2.0 \
    --train_type lora --lora_rank 16 \
    --target_modules all-linear \
    --agent_template hermes \
    --loss_scale react \
    --max_length 8192 \
    --learning_rate 1e-4 \
    --num_train_epochs 2 \
    --output_dir out/agent-sft
```

## 3. Step 2：BFCL 初评

```bash
swift eval \
    --model out/agent-sft \
    --eval_dataset bfcl_v3 \
    --eval_batch_size 16 \
    --infer_backend vllm \
    --output_dir eval/agent-sft
```

记录 baseline 分数。如果 < 60%，考虑 SFT 加数据。

## 4. Step 3：Multi-turn GRPO

```bash
swift rlhf --rlhf_type grpo \
    --model out/agent-sft \
    --dataset my-tool-rl-data \
    --external_plugins my_pkg/envs.py \
    --multi_turn true --multi_turn_func search_calc \
    --max_turns 5 \
    --reward_funcs agent_outcome agent_efficiency \
    --reward_weights 1.0 0.3 \
    --num_generations 4 \
    --train_type lora --lora_rank 16 \
    --use_vllm true \
    --output_dir out/agent-grpo
```

## 5. Step 5：合并 + 量化

```bash
# 合并 LoRA
swift export \
    --adapters out/agent-grpo \
    --merge_lora true \
    --output_dir merged/agent-7b

# AWQ 量化
swift export \
    --model merged/agent-7b \
    --quant_method awq \
    --quant_bits 4 \
    --quant_n_samples 256 \
    --output_dir merged/agent-7b-awq
```

## 6. Step 6：部署

```bash
swift deploy \
    --model merged/agent-7b-awq \
    --infer_backend vllm \
    --port 8000 \
    --tensor_parallel_size 1 \
    --max_model_len 8192 \
    --served_model_name my-agent
```

## 7. 调用示例

```python
import json
from openai import OpenAI

client = OpenAI(base_url='http://localhost:8000/v1', api_key='dummy')

tools = [{
    'type': 'function',
    'function': {
        'name': 'get_weather',
        'description': '获取某城今日天气',
        'parameters': {
            'type': 'object',
            'properties': {'city': {'type': 'string'}},
            'required': ['city'],
        },
    },
}]

messages = [{'role': 'user', 'content': '北京今天怎么样?'}]
for _ in range(5):
    resp = client.chat.completions.create(
        model='my-agent', messages=messages, tools=tools)
    msg = resp.choices[0].message
    messages.append(msg.model_dump(exclude_none=True))
    if msg.tool_calls:
        for call in msg.tool_calls:
            args = json.loads(call.function.arguments)
            obs  = my_actual_weather_api(**args)
            messages.append({
                'role': 'tool', 'tool_call_id': call.id,
                'name': call.function.name, 'content': obs,
            })
    else:
        print('Final answer:', msg.content)
        break
```
