## 3. 版本锁定与 DVC

```python
# scripts/freeze_dataset.py
"""生成数据集指纹，以防训测错位。"""
import hashlib, json, pathlib, sys

def sha256_of(path: pathlib.Path) -> str:
    h = hashlib.sha256()
    with path.open('rb') as f:
        for chunk in iter(lambda: f.read(1 << 20), b''):
            h.update(chunk)
    return h.hexdigest()

root = pathlib.Path(sys.argv[1])
fingerprint = {
    str(p.relative_to(root)): sha256_of(p)
    for p in sorted(root.rglob('*.wav')) if p.is_file()
}
(root.parent / f'{root.name}.sha256.json').write_text(
    json.dumps(fingerprint, indent=2, ensure_ascii=False)
)
print(f'frozen {len(fingerprint)} files')
```
