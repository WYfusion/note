## 0. 定位

> 一句话：实现从任意采样率到 16 / 24 kHz 的高质量重采样，附带抗混叠滤波。

---

## 1. 原理

重采样本质是一个多相多率信号处理问题。设原始采样率 $f_s$、目标采样率 $f_t$，重采样比为 $f_t / f_s = L / M$，其中 $L, M$ 互质：

$$y[n] = \sum_{k=-\infty}^{\infty} x[k] \cdot h(nM/L - k)$$

其中 $h$ 是以 $\min(f_s, f_t) / 2$ 为截止频率的抗混叠低通滤波器。打破这条质量红线则产生 mirror（镜像混叠）伪影。

```python
# audio_ops/resampler.py
"""高质量重采样与抗混叠滤波实现。

设计要点：
1. 优先 GPU（`torchaudio.transforms.Resample`）。
2. fallback 到 CPU `soxr` 处理边界采样率。
3. 以 LRU 缓存 Resample 模块避免重复初始化（kaiser 窗计算贵）。
4. 提供一句口与批量接口。
"""
from __future__ import annotations

import functools
import math
from dataclasses import dataclass
from typing import List, Optional, Sequence, Tuple

import numpy as np
import torch
import torchaudio


@dataclass(frozen=True)
class ResampleSpec:
    src_sr: int
    dst_sr: int
    rolloff: float = 0.99
    lowpass_filter_width: int = 64
    resampling_method: str = 'sinc_interp_kaiser'
    beta: float = 14.769656459379492  # kaiser-window parameter

    @property
    def gcd(self) -> int:
        return math.gcd(self.src_sr, self.dst_sr)

    @property
    def L(self) -> int:
        return self.dst_sr // self.gcd

    @property
    def M(self) -> int:
        return self.src_sr // self.gcd


@functools.lru_cache(maxsize=64)
def _build_resampler(spec: ResampleSpec, device: str) -> torchaudio.transforms.Resample:
    return torchaudio.transforms.Resample(
        orig_freq=spec.src_sr,
        new_freq=spec.dst_sr,
        rolloff=spec.rolloff,
        lowpass_filter_width=spec.lowpass_filter_width,
        resampling_method=spec.resampling_method,
        beta=spec.beta,
    ).to(device)


def resample(wav: torch.Tensor, src_sr: int, dst_sr: int) -> torch.Tensor:
    """单项重采样。wav 形状 (channels, samples) 或 (samples,)。"""
    if src_sr == dst_sr:
        return wav
    spec = ResampleSpec(src_sr=src_sr, dst_sr=dst_sr)
    device = 'cuda' if wav.is_cuda else 'cpu'
    resampler = _build_resampler(spec, device)
    if wav.dim() == 1:
        wav = wav.unsqueeze(0)
    return resampler(wav.to(device))


def batch_resample(
    wavs: Sequence[torch.Tensor],
    src_srs: Sequence[int],
    dst_sr: int,
    device: str = 'cuda',
) -> List[torch.Tensor]:
    """按同采样率分组重采样，最大限度复用 Resample 模块。"""
    by_sr: dict[int, list[int]] = {}
    for i, sr in enumerate(src_srs):
        by_sr.setdefault(sr, []).append(i)
    out: List[Optional[torch.Tensor]] = [None] * len(wavs)
    for sr, idxs in by_sr.items():
        resampler = _build_resampler(ResampleSpec(sr, dst_sr), device)
        for i in idxs:
            wav = wavs[i].to(device)
            if wav.dim() == 1:
                wav = wav.unsqueeze(0)
            out[i] = resampler(wav)
    return [w for w in out if w is not None]


# ----------------------------------------------------------
# CPU fallback (soxr)
# ----------------------------------------------------------
try:
    import soxr  # type: ignore
    HAS_SOXR = True
except ImportError:
    HAS_SOXR = False


def resample_cpu(wav: np.ndarray, src_sr: int, dst_sr: int) -> np.ndarray:
    if not HAS_SOXR:
        raise RuntimeError('pip install soxr')
    if src_sr == dst_sr:
        return wav
    return soxr.resample(wav.astype(np.float32), src_sr, dst_sr, quality='HQ')


# ----------------------------------------------------------
# Aliasing self-check
# ----------------------------------------------------------
def alias_energy(wav: np.ndarray, sr: int, target_sr: int) -> float:
    """返回超过 target_sr/2 频带的能量占比，高于 1e-3 即为明显别名。"""
    if wav.ndim > 1:
        wav = wav.mean(0)
    fft = np.fft.rfft(wav)
    freqs = np.fft.rfftfreq(len(wav), 1 / sr)
    total = float((np.abs(fft) ** 2).sum())
    above = float((np.abs(fft[freqs > target_sr / 2]) ** 2).sum())
    return above / max(total, 1e-12)


if __name__ == '__main__':
    # 自检：44.1k -> 16k
    sr0, sr1 = 44_100, 16_000
    t = torch.linspace(0, 1, sr0)
    wav = (
        torch.sin(2 * math.pi * 440 * t) +
        0.4 * torch.sin(2 * math.pi * 18_000 * t)  # 会在 16k 混叠上
    )
    out = resample(wav, sr0, sr1)
    print('out shape:', out.shape)
    e = alias_energy(out.squeeze(0).cpu().numpy(), sr1, sr1)
    assert e < 1e-3, f'aliasing detected: {e:.4e}'
    print('aliasing energy =', e)
```
