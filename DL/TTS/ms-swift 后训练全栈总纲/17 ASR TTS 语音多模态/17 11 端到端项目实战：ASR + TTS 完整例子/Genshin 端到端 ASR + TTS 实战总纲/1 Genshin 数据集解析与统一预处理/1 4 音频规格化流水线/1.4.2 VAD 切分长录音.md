## 2. 完整实现

```python
# audio_ops/vad.py
"""以 FunASR fsmn-vad 为主、Silero 为备胎的废话切分。

详细设计：
1. 输出粒度 0.6–25 s，超过在语音中间以静音点切开。
2. 在废话起止补 50 ms 衰减满足人耳连贯。
3. 以 LRU 缓存加载模型，避免多进程重复初始化。
"""
from __future__ import annotations

import functools
import math
from dataclasses import dataclass
from typing import Iterator, List, Tuple

import numpy as np
import torch


@dataclass
class VadSegment:
    start_sec: float
    end_sec: float

    @property
    def duration(self) -> float:
        return self.end_sec - self.start_sec


@functools.lru_cache(maxsize=4)
def _load_funasr(device: str = 'cuda:0'):
    from funasr import AutoModel
    return AutoModel(model='fsmn-vad', device=device)


@functools.lru_cache(maxsize=4)
def _load_silero():
    model, utils = torch.hub.load(
        repo_or_dir='snakers4/silero-vad',
        model='silero_vad', force_reload=False, trust_repo=True,
    )
    return model, utils


def _detect_funasr(audio_path: str) -> List[Tuple[float, float]]:
    model = _load_funasr()
    res = model.generate(input=audio_path)[0]
    return [(s / 1000.0, e / 1000.0) for s, e in res['value']]


def _detect_silero(wav: torch.Tensor, sr: int) -> List[Tuple[float, float]]:
    model, utils = _load_silero()
    get_speech_timestamps = utils[0]
    if sr != 16_000:
        from .resampler import resample
        wav = resample(wav, sr, 16_000)
        sr = 16_000
    ts = get_speech_timestamps(wav.squeeze(0), model, sampling_rate=sr,
                               min_silence_duration_ms=300, threshold=0.5)
    return [(t['start'] / sr, t['end'] / sr) for t in ts]


def _merge_short(
    segs: List[Tuple[float, float]],
    target_min: float = 0.6,
    target_max: float = 25.0,
    silence_pad: float = 0.05,
) -> List[VadSegment]:
    """合并过短、切分过长、补衰减。"""
    if not segs:
        return []
    merged: List[List[float]] = [list(segs[0])]
    for s, e in segs[1:]:
        if s - merged[-1][1] < 0.3 and (e - merged[-1][0]) <= target_max:
            merged[-1][1] = e
        else:
            merged.append([s, e])

    out: List[VadSegment] = []
    for s, e in merged:
        s = max(0.0, s - silence_pad)
        e = e + silence_pad
        if e - s < target_min:
            continue  # 过短丢弃
        if e - s <= target_max:
            out.append(VadSegment(s, e))
        else:
            n = math.ceil((e - s) / target_max)
            step = (e - s) / n
            for k in range(n):
                out.append(VadSegment(s + k * step, s + (k + 1) * step))
    return out


def vad_split(
    wav: torch.Tensor,
    sr: int,
    backend: str = 'funasr',
    audio_path: str | None = None,
    target_min: float = 0.6,
    target_max: float = 25.0,
) -> Iterator[torch.Tensor]:
    """返回一个生成器，逐个产出语音片段张量。"""
    if backend == 'funasr':
        if audio_path is None:
            raise ValueError('FunASR 需提供 audio_path')
        raw = _detect_funasr(audio_path)
    else:
        raw = _detect_silero(wav, sr)

    segments = _merge_short(raw, target_min=target_min, target_max=target_max)
    if wav.dim() == 1:
        wav = wav.unsqueeze(0)
    for seg in segments:
        s_sample = int(seg.start_sec * sr)
        e_sample = int(seg.end_sec * sr)
        yield wav[:, s_sample:e_sample]


if __name__ == '__main__':
    import torchaudio, sys
    wav, sr = torchaudio.load(sys.argv[1])
    if wav.shape[0] > 1:
        wav = wav.mean(0, keepdim=True)
    for i, seg in enumerate(vad_split(wav, sr, backend='silero')):
        torchaudio.save(f'/tmp/vad_{i:03d}.wav', seg, sr)
    print('done')
```

## 3. 常见坍坑

> [!important]
> 
> 1. **不补衰减** → 首末字被吃，ASR 推出缺首字。 
> 
> 1. **阈值太低** → 背景哼哼字被误识别为语音。 
> 
> 1. **未合并过短片段** → 训练 batch 中出现大量 0.2 s 片。
