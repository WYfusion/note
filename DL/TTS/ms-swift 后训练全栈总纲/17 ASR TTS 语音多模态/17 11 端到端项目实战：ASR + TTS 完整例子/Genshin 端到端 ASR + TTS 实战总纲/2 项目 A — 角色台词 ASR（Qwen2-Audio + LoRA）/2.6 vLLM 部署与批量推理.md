## 0. 定位

> 一句话：用 swift deploy + vLLM 提供 OpenAI 兼容接口，批量推理 ASR 测试集。

---

## 1. 部署命令

```bash
# scripts/deploy_asr.sh
MERGED_DIR=out/genshin-asr-merged

# 1. 合并 LoRA 到 base
swift export \
  --adapters out/genshin-asr-qwen2audio/checkpoint-final \
  --merge_lora true \
  --output_dir $MERGED_DIR

# 2. vLLM 启动
CUDA_VISIBLE_DEVICES=0 swift deploy \
  --model $MERGED_DIR \
  --infer_backend vllm \
  --max_model_len 4096 \
  --gpu_memory_utilization 0.85 \
  --port 8000
```

## 2. 批量推理客户端

```python
# infer/batch_asr.py
import asyncio, base64, json, pathlib, aiohttp, pandas as pd
from tqdm.auto import tqdm

ENDPOINT = 'http://localhost:8000/v1/chat/completions'
MODEL    = 'genshin-asr'

async def call_one(session, audio_path: str):
    audio_b64 = base64.b64encode(pathlib.Path(audio_path).read_bytes()).decode()
    payload = {
        'model': MODEL,
        'messages': [{
            'role': 'user',
            'content': [
                {'type': 'audio', 'audio': f'data:audio/wav;base64,{audio_b64}'},
                {'type': 'text',  'text':  '给出 JSON：transcription / speaker / emotion / language。'},
            ],
        }],
        'temperature': 0.0,
        'max_tokens': 256,
    }
    async with session.post(ENDPOINT, json=payload, timeout=60) as r:
        data = await r.json()
    return audio_path, data['choices'][0]['message']['content']

async def main(test_jsonl: str):
    records = [json.loads(l) for l in pathlib.Path(test_jsonl).open()]
    sem = asyncio.Semaphore(8)
    async with aiohttp.ClientSession() as s:
        async def bound(r):
            async with sem: return await call_one(s, r['audios'][0])
        results = await asyncio.gather(*[bound(r) for r in records])
    with open('preds.jsonl', 'w') as f:
        for path, raw in results:
            try: parsed = json.loads(raw.split('JSON:')[-1].strip())
            except Exception: parsed = {'transcription': raw}
            f.write(json.dumps({'audio_path': path, **{f'pred_{k}': v for k, v in parsed.items()}}) + '\n')

asyncio.run(main('data/genshin_asr_test.jsonl'))
```
