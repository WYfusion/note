## 1. 评测集构造

```python
# eval/build_testset.py
import pandas as pd
from sklearn.model_selection import GroupShuffleSplit

df = pd.read_parquet('data/clean/asr_verified.parquet')
gss = GroupShuffleSplit(test_size=0.05, n_splits=1, random_state=42)
tr_idx, te_idx = next(gss.split(df, groups=df['speaker']))
test = df.iloc[te_idx]
# 补充鲁棒性切片
robust = test.assign(
    snr_low=lambda d: d['snr_db'] < 10,
    long=lambda d: d['duration_s'] > 20,
)
robust.to_parquet('data/genshin_asr_test.parquet')
```

---

## 2. 主评测脚本骨架

```python
# eval/eval_asr.py
import json, jiwer, pandas as pd
from collections import Counter
from sklearn.metrics import f1_score, accuracy_score

ref = pd.read_parquet('data/genshin_asr_test.parquet')
pred = pd.read_json('preds.jsonl', lines=True)
df = ref.merge(pred, on='audio_path')

cer_zh = jiwer.cer(
    df.query("language == 'zh'")['transcription'].tolist(),
    df.query("language == 'zh'")['pred_transcription'].tolist(),
)
speaker_f1 = f1_score(df['speaker'], df['pred_speaker'], average='macro')
emo_acc = accuracy_score(df['emotion'], df['pred_emotion'])
print({'CER_zh': cer_zh, 'speaker_F1': speaker_f1, 'emo_acc': emo_acc})
```
