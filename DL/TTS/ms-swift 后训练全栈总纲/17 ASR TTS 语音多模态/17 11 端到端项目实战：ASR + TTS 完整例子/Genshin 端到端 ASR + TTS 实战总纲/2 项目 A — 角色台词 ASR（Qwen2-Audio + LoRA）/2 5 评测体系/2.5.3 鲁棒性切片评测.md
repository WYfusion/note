## 1. SNR 估计

以 WADA-SNR（Waveform Amplitude Distribution Analysis）估计信噪比：

```python
# eval/snr_wada.py
import numpy as np, soundfile as sf
from scipy.special import erf, erfinv

GAMMA_TABLE = np.linspace(-20, 100, 1201)

def wada_snr(wav: np.ndarray, sr: int) -> float:
    if wav.ndim > 1: wav = wav.mean(0)
    abs_wav = np.abs(wav).clip(1e-12, None)
    log_abs = np.log(abs_wav)
    g = np.mean(log_abs) - np.log(np.mean(abs_wav))  # 几何-算术比
    snr_db = float(np.interp(g, _g_for_snr_table(GAMMA_TABLE), GAMMA_TABLE))
    return snr_db

def _g_for_snr_table(snr_db_arr: np.ndarray) -> np.ndarray:
    """预计算 SNR -> g 映射表。"""
    rho = 10 ** (-snr_db_arr / 10)
    g = (
        np.log(np.exp(-rho / 2)) - 0.5 * np.log(1 + 2 * rho)
        - 0.5 * np.log(np.pi) - np.log(2)
    )
    return g

if __name__ == '__main__':
    wav, sr = sf.read('test.wav')
    print('SNR (dB):', wada_snr(wav, sr))
```

## 2. 切片评测

```python
# eval/robustness_slice.py
import pandas as pd, json
from .cer_wer import corpus_cer
from .snr_wada import wada_snr

def build_slice(df: pd.DataFrame) -> pd.DataFrame:
    df = df.copy()
    df['snr'] = df['audio_path'].map(lambda p: wada_snr(*sf.read(p)))
    df['snr_bin'] = pd.cut(df['snr'], bins=[-10, 5, 15, 25, 100],
                            labels=['<5', '5-15', '15-25', '>25'])
    df['len_bin'] = pd.cut(df['duration_s'], bins=[0, 5, 15, 25, 999],
                            labels=['<5', '5-15', '15-25', '>25'])
    return df

def report_slices(df: pd.DataFrame) -> dict:
    out = {}
    for col in ['snr_bin', 'len_bin', 'language', 'speaker_type']:
        out[col] = {}
        for k, sub in df.groupby(col, observed=True):
            out[col][str(k)] = {
                'n': int(len(sub)),
                'cer': corpus_cer(sub['transcription'].tolist(),
                                   sub['pred_transcription'].tolist()),
            }
    return out

if __name__ == '__main__':
    df = pd.read_parquet('preds.parquet')
    df = build_slice(df)
    rep = report_slices(df)
    print(json.dumps(rep, indent=2, ensure_ascii=False))
```

## 3. 阅读指标

> [!important]
> 
> **压越所谓隐藏退化**：SNR <5dB 的 CER 高于总体 1.5× 是正常；超过 3× 需补 hard-synth 样本。所有切片都需事后会看，避免「总体下降 但 某门在爆炚」场景。
