## 0. 定位

> 一句话：训练一个轻量风格分类器（wav2vec2 + linear head）验证 TTS 是否能「听起来」体现 prompt 给出的风格。

## 1. 分类器训练

```python
# eval/style_classifier.py
import torch, torchaudio, pandas as pd
from torch import nn
from transformers import AutoFeatureExtractor, Wav2Vec2Model
from torch.utils.data import Dataset, DataLoader

LABELS = ['formal', 'gentle', 'excited']


class StyleHead(nn.Module):
    def __init__(self, n_class=3, hidden=768):
        super().__init__()
        self.backbone = Wav2Vec2Model.from_pretrained('facebook/wav2vec2-base')
        for p in self.backbone.parameters():
            p.requires_grad = False
        self.head = nn.Sequential(nn.Linear(hidden, 256), nn.ReLU(), nn.Linear(256, n_class))

    def forward(self, wav, mask=None):
        feat = self.backbone(wav, attention_mask=mask).last_hidden_state.mean(1)
        return self.head(feat)


class StyleDS(Dataset):
    def __init__(self, parquet, fe):
        self.df = pd.read_parquet(parquet)
        self.fe = fe
        self.lbl = {l: i for i, l in enumerate(LABELS)}

    def __len__(self):
        return len(self.df)

    def __getitem__(self, idx):
        row = self.df.iloc[idx]
        wav, sr = torchaudio.load(row['audio_path'])
        if sr != 16_000:
            wav = torchaudio.functional.resample(wav, sr, 16_000)
        feat = self.fe(wav.mean(0).numpy(), sampling_rate=16_000, return_tensors='pt')
        return feat['input_values'][0], self.lbl[row['style']]


def train(parquet: str, out_path: str, epochs: int = 3):
    fe = AutoFeatureExtractor.from_pretrained('facebook/wav2vec2-base')
    ds = StyleDS(parquet, fe)
    loader = DataLoader(ds, batch_size=8, shuffle=True, num_workers=2)
    model = StyleHead().cuda()
    opt = torch.optim.AdamW(model.head.parameters(), lr=3e-4)
    crit = nn.CrossEntropyLoss()
    for ep in range(epochs):
        for x, y in loader:
            x, y = x.cuda(), y.cuda()
            logits = model(x)
            loss = crit(logits, y)
            loss.backward(); opt.step(); opt.zero_grad()
        print(f'epoch {ep} loss={loss.item():.3f}')
    torch.save(model.state_dict(), out_path)
```

## 2. 评测 TTS 风格准确率

```python
# eval/style_eval.py
import torch, pandas as pd
from .style_classifier import StyleHead, LABELS
from transformers import AutoFeatureExtractor
import torchaudio

def predict_style(model, fe, wav_path):
    wav, sr = torchaudio.load(wav_path)
    if sr != 16_000:
        wav = torchaudio.functional.resample(wav, sr, 16_000)
    feat = fe(wav.mean(0).numpy(), sampling_rate=16_000, return_tensors='pt')
    with torch.inference_mode():
        logits = model(feat['input_values'].cuda())
    return LABELS[logits.argmax(-1).item()]


def eval_style(synth_meta_jsonl: str, ckpt: str) -> float:
    fe = AutoFeatureExtractor.from_pretrained('facebook/wav2vec2-base')
    model = StyleHead().cuda()
    model.load_state_dict(torch.load(ckpt))
    model.eval()
    df = pd.read_json(synth_meta_jsonl, lines=True)
    correct = 0
    for _, row in df.iterrows():
        pred = predict_style(model, fe, row['audio_path'])
        correct += int(pred == row['expected_style'])
    return correct / len(df)
```
