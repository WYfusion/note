## 0. 定位

> 一句话：TTS 合成 wav 交给独立 ASR 转写，与输入文本算 WER/CER，能客观反映「说什么能听清」。

---

## 1. 评测脚本

```python
# eval/tts_wer.py
import pandas as pd, jiwer, json, pathlib
from funasr import AutoModel
from .cer_wer import normalize_text

asr_zh = AutoModel(model='paraformer-zh', vad_model='fsmn-vad', device='cuda:0')

def revcheck_tts(synth_dir: str, ref_jsonl: str) -> pd.DataFrame:
    refs = [json.loads(l) for l in pathlib.Path(ref_jsonl).open()]
    rows = []
    for ex in refs:
        wav = pathlib.Path(synth_dir) / (pathlib.Path(ex['audios'][0]).stem + '.wav')
        if not wav.exists(): continue
        hyp = asr_zh.generate(input=str(wav))[0]['text']
        ref = ex['messages'][1]['content'].split('文本: ', 1)[-1]
        ref_n, hyp_n = normalize_text(ref, 'zh'), normalize_text(hyp, 'zh')
        rows.append({
            'wav': str(wav),
            'ref': ref_n,
            'hyp': hyp_n,
            'cer': jiwer.cer(ref_n, hyp_n),
        })
    return pd.DataFrame(rows)

if __name__ == '__main__':
    df = revcheck_tts('syn/', 'data/genshin_tts_test.jsonl')
    print(f'TTS 反推 CER 均值: {df["cer"].mean():.3f}')
    print(f'CER \< 5% 样本占比: {(df["cer"] \< 0.05).mean():.2%}')
```
