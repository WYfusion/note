## 2. 原生流式服务骨架

```python
# infer/tts_server.py
import asyncio, io, soundfile as sf, torch
from fastapi import FastAPI, WebSocket
from transformers import Qwen2_5OmniProcessor, Qwen2_5OmniForConditionalGeneration

app = FastAPI()
MODEL_DIR = 'out/genshin-tts-omni-merged'
processor = Qwen2_5OmniProcessor.from_pretrained(MODEL_DIR)
model = Qwen2_5OmniForConditionalGeneration.from_pretrained(
    MODEL_DIR, torch_dtype=torch.bfloat16).to('cuda').eval()

@app.websocket('/tts')
async def tts_endpoint(ws: WebSocket):
    await ws.accept()
    payload = await ws.receive_json()  # {role, style, speed, text}
    sys = f'你是原神角色「{payload["role"]}」，按指定风格与语速朗读中文。'
    user = f'风格: {payload["style"]}\n语速: {payload["speed"]}\n文本: {payload["text"]}'
    inputs = processor.apply_chat_template([
        {'role': 'system', 'content': sys},
        {'role': 'user',   'content': user},
    ], add_generation_prompt=True, return_tensors='pt').to('cuda')
    async for chunk in model.generate_stream(**inputs, output_audio=True):
        wav = chunk['audio'].cpu().numpy()
        buf = io.BytesIO(); sf.write(buf, wav, 24000, format='WAV')
        await ws.send_bytes(buf.getvalue())
    await ws.close()
```
