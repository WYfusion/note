### 2.2 工具函数

```python
import os
import librosa
import numpy as np
import pandas as pd

def load_audio(path, sr=16000):
    """统一加载音频，重采样到目标采样率"""
    audio, _ = librosa.load(path, sr=sr)
    return audio, sr

def get_file_pairs(ref_dir, gen_dir):
    """匹配参考和生成的音频文件对"""
    ref_files = sorted(os.listdir(ref_dir))
    pairs = []
    for f in ref_files:
        gen_path = os.path.join(gen_dir, f)
        ref_path = os.path.join(ref_dir, f)
        if os.path.exists(gen_path):
            pairs.append((ref_path, gen_path))
    return pairs
```

## 三、WER 计算（内容准确性）

### 3.1 英文 WER（Faster-Whisper）

```python
from faster_whisper import WhisperModel
import jiwer

def compute_wer_en(gen_dir, text_dir):
    model = WhisperModel("large-v3", device="cuda")
    results = []
    for fname in sorted(os.listdir(gen_dir)):
        if not fname.endswith(".wav"):
            continue
        segments, _ = model.transcribe(
            os.path.join(gen_dir, fname), language="en"
        )
        hypothesis = " ".join([s.text for s in segments]).strip()
        with open(os.path.join(text_dir, fname.replace(".wav", ".txt"))) as f:
            reference = f.read().strip()
        results.append({"file": fname, "wer": jiwer.wer(reference, hypothesis)})
    df = pd.DataFrame(results)
    print(f"Average WER (EN): {df['wer'].mean():.4f}")
    return df
```

### 3.2 中文 CER（Paraformer）

```python
from funasr import AutoModel

def compute_cer_zh(gen_dir, text_dir):
    model = AutoModel(model="paraformer-zh", device="cuda")
    results = []
    for fname in sorted(os.listdir(gen_dir)):
        if not fname.endswith(".wav"):
            continue
        res = model.generate(input=os.path.join(gen_dir, fname))
        hypothesis = res[0]["text"]
        with open(os.path.join(text_dir, fname.replace(".wav", ".txt"))) as f:
            reference = f.read().strip()
        results.append({"file": fname, "cer": jiwer.cer(reference, hypothesis)})
    df = pd.DataFrame(results)
    print(f"Average CER (ZH): {df['cer'].mean():.4f}")
    return df
```

## 四、SIM 计算（说话人相似度）

```python
from resemblyzer import VoiceEncoder, preprocess_wav

def compute_sim(prompt_dir, gen_dir):
    encoder = VoiceEncoder()
    results = []
    for fname in sorted(os.listdir(gen_dir)):
        if not fname.endswith(".wav"):
            continue
        prompt_path = os.path.join(prompt_dir, fname)
        gen_path = os.path.join(gen_dir, fname)
        if not os.path.exists(prompt_path):
            continue
        prompt_emb = encoder.embed_utterance(preprocess_wav(prompt_path))
        gen_emb   = encoder.embed_utterance(preprocess_wav(gen_path))
        sim = np.dot(prompt_emb, gen_emb) / (
            np.linalg.norm(prompt_emb) * np.linalg.norm(gen_emb)
        )
        results.append({"file": fname, "sim": float(sim)})
    df = pd.DataFrame(results)
    print(f"Average SIM: {df['sim'].mean():.4f}")
    return df
```

> [!tip] 说话人编码器选择对比

> - **WavLM-TDNN**：Seed-TTS-Eval 官方使用，精度最高，推荐论文级评测

> - **ECAPA-TDNN (SpeechBrain)**：工业级精度，`speechbrain/spkrec-ecapa-voxceleb`

> - **Resemblyzer**：基于 GE2E，轻量易用，适合快速实验

## 五、PESQ + STOI（信号级质量）

> [!warning] 使用前提

> PESQ / STOI 需要参考语音与合成语音**严格时间对齐**，仅适用于 **Tokenizer 重建评测**，**不适用于**零样本克隆场景。

```python
from pesq import pesq
from pystoi import stoi

def compute_signal_metrics(ref_dir, gen_dir, sr=16000):
    pairs = get_file_pairs(ref_dir, gen_dir)
    results = []
    for ref_path, gen_path in pairs:
        ref_audio, _ = load_audio(ref_path, sr)
        gen_audio, _ = load_audio(gen_path, sr)
        min_len = min(len(ref_audio), len(gen_audio))
        ref_audio, gen_audio = ref_audio[:min_len], gen_audio[:min_len]
        try:
            pesq_score = pesq(sr, ref_audio, gen_audio, "wb")
        except Exception:
            pesq_score = float("nan")
        stoi_score = stoi(ref_audio, gen_audio, sr, extended=False)
        results.append({
            "file": os.path.basename(ref_path),
            "pesq": pesq_score, "stoi": stoi_score
        })
    df = pd.DataFrame(results)
    print(f"PESQ: {df['pesq'].mean():.3f} | STOI: {df['stoi'].mean():.4f}")
    return df
```

## 六、UTMOS（自动 MOS 预测）

```python
import torch, torchaudio

def compute_utmos(gen_dir):
    """SaruLab 官方 UTMOS 模型"""
    predictor = torch.hub.load(
        "tarepan/SpeechMOS:v1.2.0",
        "utmos22_strong",
        trust_repo=True
    )
    results = []
    for fname in sorted(os.listdir(gen_dir)):
        if not fname.endswith(".wav"):
            continue
        wav, sr = torchaudio.load(os.path.join(gen_dir, fname))
        if sr != 16000:
            wav = torchaudio.transforms.Resample(sr, 16000)(wav)
        score = predictor(wav, sr=16000).item()
        results.append({"file": fname, "utmos": score})
    df = pd.DataFrame(results)
    print(f"Average UTMOS: {df['utmos'].mean():.3f}")
    return df
```

## 七、结果聚合与可视化

```python
import matplotlib.pyplot as plt

def plot_comparison(results_dict):
    """
    多模型评测结果对比柱状图
    results_dict: {"ModelA": {"WER": 1.24, "SIM": 0.72, "UTMOS": 4.16}, ...}
    """
    models  = list(results_dict.keys())
    metrics = list(results_dict[models[0]].keys())
    colors  = ["\#4A90D9", "\#7B68EE", "\#2ECC71", "\#E74C3C"]
    
    fig, axes = plt.subplots(1, len(metrics), figsize=(4 * len(metrics), 5))
    for i, metric in enumerate(metrics):
        values = [results_dict[m][metric] for m in models]
        bars = axes[i].bar(models, values, color=colors[:len(models)])
        axes[i].set_title(metric, fontsize=14)
        for bar, val in zip(bars, values):
            axes[i].text(
                bar.get_x() + bar.get_width() / 2,
                bar.get_height() + 0.01,
                f"{val:.3f}", ha="center", fontsize=10
            )
    plt.tight_layout()
    plt.savefig("results/comparison.png", dpi=150)
    plt.show()

# 示例数据
plot_comparison({
    "Qwen3-TTS": {"WER↓": 1.24, "SIM↑": 0.721, "UTMOS↑": 4.16},

    "MaskGCT":   {"WER↓": 2.60, "SIM↑": 0.687, "UTMOS↑": 3.82},
})
```

## 八、一键完整 Pipeline

```python
def run_full_evaluation(
    gen_dir, ref_dir=None, prompt_dir=None,
    text_dir=None, lang="en"
):
    """零样本克隆 / Tokenizer 重建 一键评测"""
    results = {}

    # 1. WER / CER
    if text_dir:
        if lang == "en":
            df = compute_wer_en(gen_dir, text_dir)
            results["WER"] = df["wer"].mean()
        else:
            df = compute_cer_zh(gen_dir, text_dir)
            results["CER"] = df["cer"].mean()

    # 2. SIM（需要 prompt）
    if prompt_dir:
        df = compute_sim(prompt_dir, gen_dir)
        results["SIM"] = df["sim"].mean()

    # 3. 信号级（需要参考，仅重建场景）
    if ref_dir:
        df = compute_signal_metrics(ref_dir, gen_dir)
        results["PESQ"] = df["pesq"].mean()
        results["STOI"] = df["stoi"].mean()

    # 4. UTMOS
    df = compute_utmos(gen_dir)
    results["UTMOS"] = df["utmos"].mean()

    print("\n" + "="*40)
    for k, v in results.items():
        print(f"  {k:10s}: {v:.4f}")
    return results

# 零样本克隆
run_full_evaluation(
    gen_dir="generated/",
    prompt_dir="prompts/",
    text_dir="texts/",
    lang="en"
)
```
