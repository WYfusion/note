### 2.2 加载工具函数

```python
import os
import librosa
import soundfile as sf
import numpy as np
import pandas as pd

def load_audio(path, sr=16000):
    """统一加载音频，重采样到目标采样率"""
    audio, orig_sr = librosa.load(path, sr=sr)
    return audio, sr

def get_file_pairs(ref_dir, gen_dir):
    """匹配参考和生成的音频文件对"""
    ref_files = sorted(os.listdir(ref_dir))
    pairs = []
    for f in ref_files:
        gen_path = os.path.join(gen_dir, f)
        ref_path = os.path.join(ref_dir, f)
        if os.path.exists(gen_path):
            pairs.append((ref_path, gen_path))
    return pairs
```

## 三、WER 计算（内容准确性）

### 3.1 英文 WER（Faster-Whisper）

```python
from faster_whisper import WhisperModel
import jiwer

def compute_wer_en(gen_dir, text_dir):
    """使用 Faster-Whisper 计算英文 WER"""
    model = WhisperModel("large-v3", device="cuda")

    results = []
    for fname in sorted(os.listdir(gen_dir)):
        if not fname.endswith(".wav"):
            continue

        # ASR 转录
        segments, _ = model.transcribe(
            os.path.join(gen_dir, fname),
            language="en"
        )
        hypothesis = " ".join([s.text for s in segments]).strip()

        # 读取原始文本
        txt_path = os.path.join(
            text_dir, fname.replace(".wav", ".txt")
        )
        with open(txt_path) as f:
            reference = f.read().strip()

        # 计算 WER
        wer = jiwer.wer(reference, hypothesis)
        results.append({
            "file": fname,
            "reference": reference,
            "hypothesis": hypothesis,
            "wer": wer
        })

    df = pd.DataFrame(results)
    avg_wer = df["wer"].mean()
    print(f"Average WER (EN): {avg_wer:.4f}")
    return df
```

### 3.2 中文 CER（Paraformer）

```python
from funasr import AutoModel

def compute_cer_zh(gen_dir, text_dir):
    """使用 Paraformer 计算中文 CER"""
    model = AutoModel(
        model="paraformer-zh",
        device="cuda"
    )

    results = []
    for fname in sorted(os.listdir(gen_dir)):
        if not fname.endswith(".wav"):
            continue

        # ASR 转录
        res = model.generate(
            input=os.path.join(gen_dir, fname)
        )
        hypothesis = res[0]["text"]

        # 读取原始文本
        txt_path = os.path.join(
            text_dir, fname.replace(".wav", ".txt")
        )
        with open(txt_path) as f:
            reference = f.read().strip()

        # 字级别 CER
        cer = jiwer.cer(reference, hypothesis)
        results.append({
            "file": fname,
            "reference": reference,
            "hypothesis": hypothesis,
            "cer": cer
        })

    df = pd.DataFrame(results)
    avg_cer = df["cer"].mean()
    print(f"Average CER (ZH): {avg_cer:.4f}")
    return df
```

## 四、SIM 计算（说话人相似度）

```python
import torch
import torchaudio
from resemblyzer import VoiceEncoder, preprocess_wav

def compute_sim(prompt_dir, gen_dir):
    """计算说话人相似度（余弦相似度）"""
    encoder = VoiceEncoder()

    results = []
    for fname in sorted(os.listdir(gen_dir)):
        if not fname.endswith(".wav"):
            continue

        prompt_path = os.path.join(prompt_dir, fname)
        gen_path = os.path.join(gen_dir, fname)

        if not os.path.exists(prompt_path):
            continue

        # 提取 Speaker Embedding
        prompt_wav = preprocess_wav(prompt_path)
        gen_wav = preprocess_wav(gen_path)

        prompt_emb = encoder.embed_utterance(prompt_wav)
        gen_emb = encoder.embed_utterance(gen_wav)

        # 余弦相似度
        sim = np.dot(prompt_emb, gen_emb) / (
            np.linalg.norm(prompt_emb) * np.linalg.norm(gen_emb)
        )
        results.append({"file": fname, "sim": sim})

    df = pd.DataFrame(results)
    avg_sim = df["sim"].mean()
    print(f"Average SIM: {avg_sim:.4f}")
    return df
```


💡**说话人编码器选择**

- **WavLM-based**：Seed-TTS-Eval 官方使用，精度最高
- **Resemblyzer**：基于 GE2E，轻量易用，适合快速实验
- **ECAPA-TDNN**：SpeechBrain 提供，工业级精度

## 五、PESQ + STOI 计算（信号级质量）

```python
from pesq import pesq
from pystoi import stoi

def compute_signal_metrics(ref_dir, gen_dir, sr=16000):
    """计算 PESQ 和 STOI"""
    pairs = get_file_pairs(ref_dir, gen_dir)

    results = []
    for ref_path, gen_path in pairs:
        ref_audio, _ = load_audio(ref_path, sr=sr)
        gen_audio, _ = load_audio(gen_path, sr=sr)

        # 对齐长度
        min_len = min(len(ref_audio), len(gen_audio))
        ref_audio = ref_audio[:min_len]
        gen_audio = gen_audio[:min_len]

        # PESQ (宽带模式)
        try:
            pesq_score = pesq(sr, ref_audio, gen_audio, "wb")
        except Exception:
            pesq_score = float("nan")

        # STOI
        stoi_score = stoi(
            ref_audio, gen_audio, sr, extended=False
        )

        results.append({
            "file": os.path.basename(ref_path),
            "pesq": pesq_score,
            "stoi": stoi_score
        })

    df = pd.DataFrame(results)
    print(f"Average PESQ: {df['pesq'].mean():.3f}")
    print(f"Average STOI: {df['stoi'].mean():.4f}")
    return df
```


⚠️**注意**：PESQ 和 STOI 需要参考语音与合成语音**严格时间对齐**。适用于 Tokenizer 重建评测，**不适用于**零样本克隆评测（因为合成内容与参考不同）。

## 六、UTMOS 计算（自动 MOS 预测）

```python
# 方案一：使用 speechmos 库
import speechmos

def compute_utmos(gen_dir):
    """批量计算 UTMOS 分数"""
    predictor = speechmos.UTMOS(device="cuda")

    results = []
    for fname in sorted(os.listdir(gen_dir)):
        if not fname.endswith(".wav"):
            continue

        score = predictor.predict(
            os.path.join(gen_dir, fname)
        )
        results.append({"file": fname, "utmos": score})

    df = pd.DataFrame(results)
    avg_utmos = df["utmos"].mean()
    print(f"Average UTMOS: {avg_utmos:.3f}")
    return df

# 方案二：使用 torch hub (SaruLab 官方)
import torch

def compute_utmos_official(gen_dir):
    """使用 SaruLab 官方 UTMOS 模型"""
    predictor = torch.hub.load(
        "tarepan/SpeechMOS:v1.2.0",
        "utmos22_strong",
        trust_repo=True
    )

    results = []
    for fname in sorted(os.listdir(gen_dir)):
        if not fname.endswith(".wav"):
            continue

        wav, sr = torchaudio.load(
            os.path.join(gen_dir, fname)
        )
        if sr != 16000:
            wav = torchaudio.transforms.Resample(
                sr, 16000
            )(wav)

        score = predictor(wav, sr=16000).item()
        results.append({"file": fname, "utmos": score})

    df = pd.DataFrame(results)
    print(f"Average UTMOS: {df['utmos'].mean():.3f}")
    return df
```

## 七、结果聚合与可视化

```python
import matplotlib.pyplot as plt
import matplotlib
matplotlib.rcParams["font.family"] = "SimHei"  # 中文显示

def aggregate_results(
    wer_df, sim_df, signal_df=None, utmos_df=None
):
    """聚合所有评测结果"""
    merged = wer_df[["file", "wer"]].copy()
    merged = merged.merge(
        sim_df[["file", "sim"]], on="file", how="outer"
    )
    if signal_df is not None:
        merged = merged.merge(
            signal_df[["file", "pesq", "stoi"]],
            on="file", how="outer"
        )
    if utmos_df is not None:
        merged = merged.merge(
            utmos_df[["file", "utmos"]],
            on="file", how="outer"
        )
    return merged

def plot_comparison(results_dict):
    """多模型评测结果对比图

    Args:
        results_dict: {"ModelA": {"WER": 1.24, "SIM": 0.72, ...}, ...}
    """
    models = list(results_dict.keys())
    metrics = list(results_dict[models[0]].keys())

    fig, axes = plt.subplots(
        1, len(metrics), figsize=(4*len(metrics), 5)
    )

    for i, metric in enumerate(metrics):
        values = [results_dict[m][metric] for m in models]
        bars = axes[i].bar(models, values, color=[
            "#4A90D9", "#7B68EE", "#2ECC71", "#E74C3C"
        ][:len(models)])
        axes[i].set_title(metric, fontsize=14)
        axes[i].set_ylabel("Score")

        # 标注数值
        for bar, val in zip(bars, values):
            axes[i].text(
                bar.get_x() + bar.get_width()/2,
                bar.get_height() + 0.01,
                f"{val:.3f}",
                ha="center", va="bottom", fontsize=10
            )

    plt.tight_layout()
    plt.savefig("results/comparison.png", dpi=150)
    plt.show()

# 示例：多模型对比
```

## 八、完整评测 Pipeline 脚本

```python
def run_full_evaluation(
    gen_dir, ref_dir=None, prompt_dir=None,
    text_dir=None, lang="en"
):
    """一键运行完整评测流程"""
    results = {}

    # 1. WER/CER
    if text_dir:
        if lang == "en":
            wer_df = compute_wer_en(gen_dir, text_dir)
            results["WER"] = wer_df["wer"].mean()
        else:
            cer_df = compute_cer_zh(gen_dir, text_dir)
            results["CER"] = cer_df["cer"].mean()

    # 2. SIM
    if prompt_dir:
        sim_df = compute_sim(prompt_dir, gen_dir)
        results["SIM"] = sim_df["sim"].mean()

    # 3. PESQ + STOI (仅重建场景)
    if ref_dir:
        signal_df = compute_signal_metrics(
            ref_dir, gen_dir
        )
        results["PESQ"] = signal_df["pesq"].mean()
        results["STOI"] = signal_df["stoi"].mean()

    # 4. UTMOS
    utmos_df = compute_utmos(gen_dir)
    results["UTMOS"] = utmos_df["utmos"].mean()

    # 5. 输出汇总
    print("\n" + "="*50)
    print("评测结果汇总")
    print("="*50)
    for k, v in results.items():
        print(f"  {k}: {v:.4f}")

    return results

# === 零样本克隆评测 ===
results_zs = run_full_evaluation(
    gen_dir="generated/",
    prompt_dir="prompts/",
    text_dir="texts/",
    lang="en"
)

# === Tokenizer 重建评测 ===
results_recon = run_full_evaluation(
    gen_dir="reconstructed/",
    ref_dir="reference/",
    text_dir="texts/",
    lang="en"
)
```
