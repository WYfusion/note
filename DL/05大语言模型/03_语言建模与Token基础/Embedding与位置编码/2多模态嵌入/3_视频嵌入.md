# 视频嵌入：时序视觉特征与动态内容理解

> [!abstract] 摘要
> 视频嵌入将视频的时空信息编码为语义向量，实现视频搜索、内容理解、动作识别等核心功能。从传统的 3D CNN 到现代的 Video-Transformer、Video-CLIP，视频嵌入技术已从静态帧分析发展到时空联合建模，为视频理解、内容推荐、视频检索等应用提供强大的技术支撑。

## 0. 统一概念：什么是视频嵌入？

> [!important] 视频嵌入的双重特性
> 视频嵌入需要同时处理**空间维度**（图像内容）和**时间维度**（帧间变化），将其编码为能够表示视频语义的向量。与图像和文本不同，视频嵌入的挑战在于：

### 视频嵌入的独特挑战

| 挑战 | 解决方案 | 技术要点 |
|------|----------|----------|
| **时空关系建模** | 3D CNN、时空注意力 | 同时建模空间和时间特征 |
| **长序列处理** | 分段编码、Transformer | 处理长视频序列 |
| **多尺度信息** | 多尺度特征融合 | 同时关注局部和全局信息 |
| **动态内容理解** | 动作识别、时序对比 | 理解视频中的动作变化 |

### 视频嵌入的应用场景

| 应用类型 | 输入形式 | 输出内容 | 核心技术 |
|----------|----------|----------|----------|
| **视频检索** | 视频片段 → 相似视频 | 语义相似视频 | 时空嵌入对比 |
| **动作识别** | 视频序列 → 动作类别 | 行为分类 | 时序特征学习 |
| **视频摘要** | 长视频 → 关键片段 | 内容摘要 | 重要性评分 |
| **视频问答** | 视频+问题 → 答案 | 多模态理解 | 图文视频对齐 |

### 1.1 传统视频特征提取

> [!note] 从 2D CNN 到 3D CNN 的演进

```python
import torch
import torch.nn as nn
import torchvision.models as models
from torchvision import transforms

class VideoFeatureExtractor:
    def __init__(self, backbone='resnet'):
        """初始化视频特征提取器"""
        self.backbone = backbone
        self.frame_extractor = None
        self.temporal_model = None

    def extract_frame_features(self, frame):
        """提取单帧特征"""
        # 使用预训练的 2D CNN
        resnet = models.resnet50(pretrained=True)
        resnet.eval()

        # 移除最后的分类层
        frame_features = nn.Sequential(*list(resnet.children())[:-1])
        with torch.no_grad():
            features = frame_features(frame)

        return features

    def extract_temporal_features(self, frame_sequence):
        """提取时序特征"""
        batch_size, seq_len, C, H, W = frame_sequence.shape

        # 重塑为批处理形式
        frames = frame_sequence.view(batch_size * seq_len, C, H, W)

        # 提取每帧特征
        all_features = []
        for i in range(0, batch_size * seq_len, 32):  # 分批处理
            batch_frames = frames[i:i+32]
            with torch.no_grad():
                features = self.extract_frame_features(batch_frames)
            all_features.append(features)

        # 恢复时间维度
        temporal_features = torch.cat(all_features, dim=0)
        temporal_features = temporal_features.view(batch_size, seq_len, -1)

        return temporal_features

    # 使用 LSTM 模拟时序关系
    class TemporalModel(nn.Module):
        def __init__(self, input_size=2048, hidden_size=512):
            super().__init__()
            self.lstm = nn.LSTM(input_size, hidden_size, batch_first=True)
            self.fc = nn.Linear(hidden_size, input_size)

        def forward(self, x):
            # x: (batch, seq_len, features)
            lstm_out, _ = self.lstm(x)
            output = self.fc(lstm_out)
            return output.mean(dim=1)  # 平均池化

# 使用示例
extractor = VideoFeatureExtractor()
temporal_model = extractor.TemporalModel()

# 处理视频帧序列
video_frames = torch.randn(1, 16, 3, 224, 224)  # (batch, frames, C, H, W)
frame_features = extractor.extract_temporal_features(video_frames)
video_embedding = temporal_model(frame_features)
```

```python
# 数据预处理
def preprocess_video(video_path, target_frames=16, target_size=(224, 224)):
    """预处理视频数据"""
    cap = cv2.VideoCapture(video_path)
    frames = []

    while len(frames) < target_frames:
        ret, frame = cap.read()
        if not ret:
            break

        # 调整大小
        frame = cv2.resize(frame, target_size)
        # BGR to RGB
        frame = cv2.cvtColor(frame, cv2.COLOR_BGR2RGB)
        frames.append(frame)

    # 填充或截断到目标帧数
    if len(frames) < target_frames:
        # 重复最后一帧
        frames.extend([frames[-1]] * (target_frames - len(frames)))
    else:
        frames = frames[:target_frames]

    # 转换为 tensor
    video_tensor = torch.tensor(frames).permute(0, 3, 1, 2)  # (frames, C, H, W)
    video_tensor = video_tensor.unsqueeze(0)  # (batch, frames, C, H, W)

    return video_tensor.float() / 255.0
```

### 1.3 光流特征提取

> [!tip] 运动信息的光流表示

```python
import cv2
import numpy as np

class OpticalFlowExtractor:
    def __init__(self):
        """初始化光流提取器"""
        # 使用 Farneback 算法
        self.prev_gray = None

    def extract_optical_flow(self, video_path):
        """提取光流特征"""
        cap = cv2.VideoCapture(video_path)
        optical_flows = []

        while True:
            ret, frame = cap.read()
            if not ret:
                break

            gray = cv2.cvtColor(frame, cv2.COLOR_BGR2GRAY)

            if self.prev_gray is not None:
                # 计算光流
                flow = cv2.calcOpticalFlowFarneback(
                    self.prev_gray, gray, None,
                    pyr_scale=0.5, levels=3,
                    winsize=15, iterations=3,
                    poly_n=5, poly_sigma=1.2,
                    flags=0
                )

                # 计算光流幅度
                magnitude, angle = cv2.cartToPolar(flow[..., 0], flow[..., 1])
                optical_flows.append(magnitude)

            self.prev_gray = gray

        cap.release()
        return np.stack(optical_flows, axis=0)

    def combine_motion_features(self, frame_features, optical_flows):
        """结合运动和静态特征"""
        # 将光流信息叠加到帧特征上
        combined_features = []

        for i, (frame_feat, flow) in enumerate(zip(frame_features, optical_flows)):
            # 扩展光流通道
            flow_expanded = np.expand_dims(flow, axis=-1)
            flow_expanded = np.repeat(flow_expanded, 3, axis=-1)  # 复制到3通道

            # 融合特征
            combined = np.concatenate([frame_feat, flow_expanded], axis=-1)
            combined_features.append(combined)

        return np.stack(combined_features, axis=0)
```

### 2.1 Video-CLIP: 图视频对齐框架

> [!important] Video-CLIP 的双流架构

```python
from transformers import AutoProcessor, AutoModel

class VideoCLIPProcessor:
    def __init__(self, model_name="openai/clip-vit-base-patch32"):
        """初始化 Video-CLIP 处理器"""
        self.processor = AutoProcessor.from_pretrained(model_name)
        self.model = AutoModel.from_pretrained(model_name)

    def encode_video_text(self, video_frames, text_queries):
        """编码视频和文本"""
        # 处理视频帧
        inputs = self.processor(
            text=text_queries,
            images=video_frames,  # list of PIL images
            return_tensors="pt",
            padding=True
        )

        # 前向传播
        outputs = self.model(**inputs)

        # 提取特征
        video_features = outputs.image_embeds
        text_features = outputs.text_embeds

        return video_features, text_features

    def video_text_similarity(self, video_frames, text_query):
        """计算视频文本相似度"""
        video_features, text_features = self.encode_video_text(
            video_frames, [text_query]
        )

        # 计算相似度
        similarity = torch.cosine_similarity(video_features, text_features, dim=-1)
        return similarity.item()

# 视频片段处理
def extract_video_clips(video_path, clip_length=16, overlap=8):
    """将长视频分割成片段"""
    cap = cv2.VideoCapture(video_path)
    clips = []

    while True:
        frames = []
        for _ in range(clip_length):
            ret, frame = cap.read()
            if not ret:
                break
            frame = cv2.cvtColor(frame, cv2.COLOR_BGR2RGB)
            frames.append(Image.fromarray(frame))

        if len(frames) < clip_length:
            break

        clips.append(frames)

        # 跳过重叠部分
        for _ in range(overlap):
            ret, _ = cap.read()
            if not ret:
                break

    cap.release()
    return clips
```

### 2.2 TimeSformer: 纯时序注意力

> [!note] TimeSformer 的分块注意力机制

```python
from transformers import TimeSformerForVideoClassification, TimeSformerImageProcessor

class TimeSformerModel:
    def __init__(self, model_name="facebook/timesformer-base-finetuned-k400"):
        """初始化 TimeSformer"""
        self.processor = TimeSformerImageProcessor.from_pretrained(model_name)
        self.model = TimeSformerForVideoClassification.from_pretrained(model_name)

    def extract_video_embedding(self, video_frames):
        """提取视频嵌入"""
        # 预处理
        inputs = self.processor(video_frames, return_tensors="pt")

        # 提取特征
        with torch.no_grad():
            outputs = self.model(**inputs, output_hidden_states=True)

        # 使用最后一层隐藏状态
        video_embedding = outputs.hidden_states[-1].mean(dim=1)
        return video_embedding

    # TimeSformer 的分块注意力实现
    class PatchEmbedding(nn.Module):
        def __init__(self, patch_size=16, embed_dim=768):
            super().__init__()
            self.patch_size = patch_size
            self.proj = nn.Conv2d(3, embed_dim, kernel_size=patch_size, stride=patch_size)

        def forward(self, x):
            # x: (batch, C, T, H, W)
            B, C, T, H, W = x.shape
            # 重塑为批处理形式
            x = x.permute(0, 2, 1, 3, 4)  # (B, T, C, H, W)
            x = x.reshape(-1, C, H, W)  # (B*T, C, H, W)

            patches = self.proj(x)  # (B*T, embed_dim, H//patch_size, W//patch_size)
            patches = patches.flatten(2).transpose(1, 2)  # (B*T, num_patches, embed_dim)

            # 恢复时间维度
            patches = patches.view(B, T, -1, patches.shape[-1])  # (B, T, num_patches, embed_dim)
            return patches
```

### 2.3 VideoMAE: 自监督预训练

> [!important] VideoMAE 的掩码自编码器

```python
class VideoMAE:
    def __init__(self, patch_size=16, mask_ratio=0.75):
        """初始化 VideoMAE"""
        self.patch_size = patch_size
        self.mask_ratio = mask_ratio

    def create_mask(self, video_shape):
        """创建掩码"""
        batch_size, time, height, width = video_shape

        # 计算每个位置的索引
        num_patches_height = height // self.patch_size
        num_patches_width = width // self.patch_size
        total_patches = num_patches_height * num_patches_width

        # 随机选择要掩码的 patch
        num_masked = int(total_patches * self.mask_ratio)
        masked_indices = torch.randperm(total_patches)[:num_masked]

        # 创建掩码张量
        mask = torch.zeros(total_patches)
        mask[masked_indices] = 1

        # 重塑为空间维度
        mask = mask.view(num_patches_height, num_patches_width)

        # 扩展到时间维度
        mask = mask.unsqueeze(0).expand(batch_size, time, -1, -1)

        return mask

    def reconstruct_patches(self, masked_patches, encoder, decoder):
        """重建被掩码的 patch"""
        # 编码可见部分
        visible_mask = (masked_patches != 0).float()
        encoded = encoder(masked_patches * visible_mask)

        # 解码
        reconstructed = decoder(encoded)

        # 计算重建损失
        loss = torch.nn.MSELoss()(reconstructed * visible_mask, masked_patches * visible_mask)

        return loss, reconstructed

# VideoMAE 的数据预处理
def preprocess_videomae(video_tensor):
    """VideoMAE 预处理"""
    # 转换为 patch 格式
    B, C, T, H, W = video_tensor.shape
    num_patches_h = H // 16
    num_patches_w = W // 16

    # 重塑为 patch
    patches = video_tensor.reshape(B, C, T, num_patches_h, 16, num_patches_w, 16)
    patches = patches.permute(0, 2, 1, 3, 4, 5, 6)  # (B, T, C, pH, pW, 16, 16)
    patches = patches.flatten(2, 4).flatten(-2)  # (B, T, C*pH*pW, 16*16)
    patches = patches.permute(0, 2, 1, 3)  # (B, C*pH*pW, T, 16*16)

    return patches
```

```python
class VideoQuestionAnswering:
    def __init__(self, model_name="videoclip"):
        """初始化视频问答系统"""
        self.model = VideoCLIPProcessor(model_name)

    def answer_question(self, video_path, question):
        """回答关于视频的问题"""
        # 加载视频帧
        video_frames = extract_video_clips(video_path)

        # 生成候选答案
        possible_answers = self.generate_candidate_answers(question)

        # 计算每个答案的概率
        answer_probs = []
        for answer in possible_answers:
            sim = self.model.video_text_similarity(video_frames, answer)
            answer_probs.append((answer, sim))

        # 选择最可能的答案
        answer_probs.sort(key=lambda x: x[1], reverse=True)
        return answer_probs[0][0]

    def generate_candidate_answers(self, question):
        """生成候选答案"""
        # 这里可以集成更大的模型来生成候选答案

            return ["红色", "蓝色", "绿色", "黄色", "黑色"]
        elif "动作" in question:
            return ["跑步", "走路", "跳跃", "坐着", "站着"]
        else:
            return ["是", "否", "不知道"]
```

### 4.1 长视频处理策略

> [!important] 分段与聚合策略

```python
class LongVideoProcessor:
    def __init__(self, clip_length=16, overlap=8):
        """初始化长视频处理器"""
        self.clip_length = clip_length
        self.overlap = overlap

    def process_long_video(self, video_path, model):
        """处理长视频"""
        # 1. 分段处理
        clips = self.extract_clips(video_path)
        all_embeddings = []

        # 2. 批量处理
        for i, clip in enumerate(clips):
            embedding = model.extract_embedding(clip)
            all_embeddings.append(embedding)

        # 3. 聚合嵌入
        video_embedding = self.aggregate_embeddings(all_embeddings)
        return video_embedding

    def aggregate_embeddings(self, embeddings, method='mean'):
        """聚合多个片段的嵌入"""
        embeddings = torch.stack(embeddings)

        if method == 'mean':
            return embeddings.mean(dim=0)
        elif method == 'max':
            return embeddings.max(dim=0)[0]
        elif method == 'attention':
            return self.attention_aggregate(embeddings)
        else:
            raise ValueError(f"Unsupported aggregation method: {method}")

    def attention_aggregate(self, embeddings):
        """注意力聚合"""
        # 计算注意力权重
        attention_weights = torch.softmax(torch.randn(embeddings.shape[1]), dim=0)

        # 加权平均
        aggregated = torch.sum(embeddings * attention_weights.unsqueeze(0), dim=1)
        return aggregated

    # 优化的内存管理
    def process_large_video(self, video_path, model, max_memory=4):
        """处理超长视频的内存优化"""
        cap = cv2.VideoCapture(video_path)
        total_frames = int(cap.get(cv2.CAP_PROP_FRAME_COUNT))

        # 分块处理
        batch_size = 100  # 每批处理的帧数
        all_embeddings = []

        start_frame = 0
        while start_frame < total_frames:
            # 读取下一批帧
            end_frame = min(start_frame + batch_size, total_frames)
            frames = []

            for _ in range(start_frame, end_frame):
                ret, frame = cap.read()
                if ret:
                    frame = cv2.cvtColor(frame, cv2.COLOR_BGR2RGB)
                    frames.append(frame)

            if frames:
                # 处理这一批
                embedding = model.extract_embedding(frames)
                all_embeddings.append(embedding)

            start_frame = end_frame

        cap.release()

        # 最终聚合
        return self.aggregate_embeddings(all_embeddings)
```

```python
class RealTimeVideoProcessor:
    def __init__(self, model, buffer_size=32):
        """初始化实时视频处理器"""
        self.model = model
        self.buffer_size = buffer_size
        self.frame_buffer = deque(maxlen=buffer_size)
        self.processing_queue = queue.Queue()
        self.is_processing = False

    def start_processing(self, video_source):
        """开始实时处理"""
        # 启动处理线程
        self.is_processing = True
        processing_thread = threading.Thread(target=self.process_frames)
        processing_thread.start()

        # 读取视频帧
        cap = cv2.VideoCapture(video_source)
        while self.is_processing:
            ret, frame = cap.read()
            if not ret:
                break

            # 转换为 PIL 图像
            frame_rgb = cv2.cvtColor(frame, cv2.COLOR_BGR2RGB)
            pil_frame = Image.fromarray(frame_rgb)

            # 添加到缓冲区
            self.frame_buffer.append(pil_frame)

            # 如果缓冲区满了，开始处理
            if len(self.frame_buffer) == self.buffer_size:
                self.processing_queue.put(list(self.frame_buffer))
                self.frame_buffer.clear()

        cap.release()
        self.is_processing = False

        """处理帧队列"""

                frames = self.processing_queue.get()
```

```python
class MultimodalVideoAnalyzer:
    def __init__(self):
        """初始化多模态视频分析器"""
        self.video_model = VideoCLIPProcessor()
        self.audio_model = AudioCLIPProcessor()
        self.text_model = TextEmbeddingModel()

    def analyze_multimodal(self, video_path, audio_path, text_context=None):
        """多模态视频分析"""
        results = {}

        # 1. 视频分析
        video_results = self.analyze_video(video_path)
        results['video'] = video_results

        # 2. 音频分析
        if audio_path:
            audio_results = self.analyze_audio(audio_path)
            results['audio'] = audio_results

        # 3. 文本分析
        if text_context:
            text_results = self.analyze_text(text_context)
            results['text'] = text_results

        # 4. 跨模态融合
        fused_results = self.fuse_multimodal_results(results)
        results['fused'] = fused_results

        return results

    def fuse_multimodal_results(self, results):
        """融合多模态结果"""
        # 视频音频对齐分析
        video_audio_alignment = self.align_video_audio(
            results['video']['features'],
            results['audio']['features']
        )

        # 视频文本对齐分析
        video_text_alignment = self.align_video_text(
            results['video']['features'],
            results['text']['features']
        )

        # 综合评分
        overall_score = self.calculate_overall_score(
            video_audio_alignment,
            video_text_alignment
        )

        return {
            'video_audio_alignment': video_audio_alignment,
            'video_text_alignment': video_text_alignment,
            'overall_score': overall_score
        }
```
