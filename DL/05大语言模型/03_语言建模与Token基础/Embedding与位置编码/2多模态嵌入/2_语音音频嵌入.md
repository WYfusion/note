# 语音音频嵌入：声音特征提取与音频文本对齐

> [!abstract] 摘要
> 语音音频嵌入将声音、语音、音乐等听觉信息编码为语义向量，实现语音识别、音频检索、音乐推荐等核心功能。从传统的声学特征提取到现代的 Wav2Vec、Whisper 等预训练模型，音频嵌入技术已实现从"特征工程"到"预训练模型"的范式转变，为多模态 AI 系统提供听觉理解能力。

## 0. 统一概念：什么是语音音频嵌入？

> [!important] 音频嵌入的本质与挑战
> 语音音频嵌入是将**声音波形**编码为**语义向量**的过程，使得语义相似的音频在向量空间中距离相近。与视觉和文本相比，音频嵌入面临独特挑战：

### 音频嵌入的特殊性

| 挑战 | 特点 | 解决方案 |
|------|------|----------|
| **时序特性** | 声音是连续的时序信号 | 递归神经网络、Transformer |
| **背景噪声** | 环境噪声干扰 | 预增强、频谱图增强 |
| **说话人差异** | 不同人的声音特征 | 说话人自适应、x-vector |
| **语速变化** | 语速快慢不一 | 时间拉伸、动态时间规整 |

### 音频嵌入的应用场景

| 应用类型 | 输入模态 | 输出形式 | 核心技术 |
|----------|----------|----------|----------|
| **语音识别** | 音频 → 文本 | 转写文本 | CTC, Attention |
| **说话人识别** | 音频 → 标签 | 说话人ID | i-vector, x-vector |
| **音频检索** | 音频 → 音频 | 相似音频片段 | 声纹匹配、语义相似 |
| **音乐推荐** | 音频 → 向量 | 兴趣偏好 | 音乐特征学习 |

## 1. 音频特征提取基础

### 1.1 传统声学特征

> [!tip] 从 MFCC 到 Chroma 特征

```python
import librosa
import numpy as np
from scipy import signal

def extract_audio_features(audio_path, sr=16000):
    """提取传统声学特征"""
    y, sr = librosa.load(audio_path, sr=sr)

    # 1. MFCC (梅尔频率倒谱系数)
    mfcc = librosa.feature.mfcc(y=y, sr=sr, n_mfcc=13)
    mfcc_delta = librosa.feature.delta(mfcc)
    mfcc_delta2 = librosa.feature.delta(mfcc, order=2)

    # 2. 频谱特征
    spectral_centroid = librosa.feature.spectral_centroid(y=y, sr=sr)
    spectral_rolloff = librosa.feature.spectral_rolloff(y=y, sr=sr)
    spectral_bandwidth = librosa.feature.spectral_bandwidth(y=y, sr=sr)
    zero_crossing_rate = librosa.feature.zero_crossing_rate(y)

    # 3. Chroma 特征 (音乐调性)
    chroma = librosa.feature.chroma_stft(y=y, sr=sr)

    # 4. 过零率 (语音检测)
    zcr = librosa.feature.zero_crossing_rate(y)

    return {
        'mfcc': mfcc,
        'mfcc_delta': mfcc_delta,
        'chroma': chroma,
        'spectral_centroid': spectral_centroid,
        'zcr': zcr
    }

# 特征拼接
def concatenate_features(features, n_mfcc=13):
    """将特征拼接成向量"""
    feature_vector = []

    # MFCC (13 + 13 + 13 = 39)
    feature_vector.extend(features['mfcc'].flatten())
    feature_vector.extend(features['mfcc_delta'].flatten())
    feature_vector.extend(features['mfcc_delta2'].flatten())

    # 其他特征 (平均池化)
    for name in ['chroma', 'spectral_centroid']:
        feat = features[name].mean(axis=1)
        feature_vector.extend(feat)

    # ZCR (原始值)
    feature_vector.extend(features['zcr'].flatten())

    return np.array(feature_vector)
```

import librosa.display

def create_spectrogram(audio_path, sr=16000):
    """创建频谱图"""
    y, sr = librosa.load(audio_path, sr=sr)

### 1.3 音频数据增强

> [!warning] 数据增强的注意事项

```python
import torch
import torchaudio
import random
import numpy as np

class AudioAugmentation:
    def __init__(self, sr=16000):
        self.sr = sr

    def add_noise(self, audio, noise_factor=0.005):
        """添加噪声"""
        noise = np.random.randn(len(audio))
        augmented_audio = audio + noise_factor * noise
        augmented_audio = augmented_audio.astype(type(audio[0]))
        return augmented_audio

    def time_shift(self, audio, shift_max=0.2):
        """时间偏移"""
        shift = int(self.sr * random.uniform(-shift_max, shift_max))
        return np.roll(audio, shift)

    def pitch_shift(self, audio, n_steps=4):
        """音高变换"""
        return librosa.effects.pitch_shift(audio, self.sr, n_steps=n_steps)

    def time_stretch(self, audio, rate=1.0):
        """时间拉伸"""
        return librosa.effects.time_stretch(audio, rate=rate)

    def apply_random_augmentation(self, audio, augmentations=['noise', 'shift', 'pitch', 'stretch']):
        """随机应用增强"""
        aug = AudioAugmentation()

        for aug_type in augmentations:
            if random.random() < 0.5:  # 50% 概率应用
                if aug_type == 'noise':
                    audio = aug.add_noise(audio)
                elif aug_type == 'shift':
                    audio = aug.time_shift(audio)
                elif aug_type == 'pitch':
                    audio = aug.pitch_shift(audio)
                elif aug_type == 'stretch':
                    audio = aug.time_stretch(audio)

        return audio
```

### 2.1 Wav2Vec 2.0: 无监督预训练革命

> [!note] Wav2Vec 2.0 的掩码预测机制

```python
import torch
import torch.nn as nn
from transformers import Wav2Vec2Processor, Wav2Vec2Model

# 加载 Wav2Vec 2.0
processor = Wav2Vec2Processor.from_pretrained("facebook/wav2vec2-base-960h")
model = Wav2Vec2Model.from_pretrained("facebook/wav2vec2-base-960h")

def extract_wav2vec_features(audio_path):
    """提取 Wav2Vec 2.0 特征"""
    # 加载音频
    audio, sr = torchaudio.load(audio_path)
    if sr != 16000:
        audio = torchaudio.transforms.Resample(sr, 16000)(audio)

    # 预处理
    input_values = processor(audio.squeeze(), return_tensors="pt", padding=True).input_values

    # 提取特征
    with torch.no_grad():
        outputs = model(input_values)

    # 使用隐藏状态作为嵌入
    embeddings = outputs.last_hidden_state
    return embeddings.mean(dim=1).squeeze()  # 平均池化

# Wav2Vec 2.0 的掩码预测
class Wav2Vec2Masked(nn.Module):
    def __init__(self, hidden_size=768, vocab_size=32):
        super().__init__()
        self.quantizer = nn.Linear(hidden_size, vocab_size)
        self.mask_token = nn.Parameter(torch.zeros(hidden_size))

    def forward(self, hidden_states):
        # 随机掩码部分 token
        mask = torch.rand_like(hidden_states) > 0.8
        masked_hidden = hidden_states.clone()
        masked_hidden[mask] = self.mask_token

        # 预测被掩码的 token
        logits = self.quantizer(masked_hidden)
        return logits
```

### 2.2 Whisper: 多语言语音识别

> [!important] Whisper 的编码器-解码器架构

```python
from transformers import WhisperProcessor, WhisperForConditionalGeneration

# 加载 Whisper
processor = WhisperProcessor.from_pretrained("openai/whisper-base")
model = WhisperForConditionalGeneration.from_pretrained("openai/whisper-base")

def transcribe_with_whisper(audio_path):
    """使用 Whisper 进行语音转写"""
    # 加载音频
    audio, sr = torchaudio.load(audio_path)

    # 预处理
    input_features = processor(audio.squeeze(), return_tensors="pt").input_features

    # 生成转录
    predicted_ids = model.generate(input_features)
    transcription = processor.batch_decode(predicted_ids, skip_special_tokens=True)

    return transcription[0]

# 音频嵌入提取
def extract_whisper_embedding(audio_path):
    """提取 Whisper 音频嵌入"""
    audio, sr = torchaudio.load(audio_path)
    input_features = processor(audio.squeeze(), return_tensors="pt").input_features

    # 获取编码器输出
    with torch.no_grad():
        encoder_outputs = model.model.encoder(input_features)

    # 使用最后一层隐藏状态作为嵌入
    embeddings = encoder_outputs.last_hidden_state
    return embeddings.mean(dim=1).squeeze()

# 多语言支持
def multilingual_transcribe(audio_path, language="zh"):
    """多语言转写"""
    audio, sr = torchaudio.load(audio_path)
    input_features = processor(audio.squeeze(), return_tensors="pt").input_features

    # 指定语言
    forced_decoder_ids = processor.get_decoder_prompt_ids(language=language)

    # 生成转录
    predicted_ids = model.generate(input_features, forced_decoder_ids=forced_decoder_ids)
    transcription = processor.batch_decode(predicted_ids, skip_special_tokens=True)

    return transcription[0]
```

### 2.3 HuBERT: 基于单元的预训练

> [!tip] HuBERT 的离散单元机制

```python
from transformers import Wav2Vec2Processor, Wav2Vec2Model

# HuBERT 使用与 Wav2Vec 相同的架构，但训练目标不同
processor = Wav2Vec2Processor.from_pretrained("facebook/hubert-base-ls960")
model = Wav2Vec2Model.from_pretrained("facebook/hubert-base-ls960")

def extract_hubert_features(audio_path):
    """提取 HuBERT 特征"""
    # 加载音频
    audio, sr = torchaudio.load(audio_path)
    if sr != 16000:
        audio = torchaudio.transforms.Resample(sr, 16000)(audio)

    # 预处理
    input_values = processor(audio.squeeze(), return_tensors="pt", padding=True).input_values

    # 提取特征
    with torch.no_grad():
        outputs = model(input_values)

    # HuBERT 使用对比学习，返回隐藏状态
    hidden_states = outputs.last_hidden_state
    return hidden_states.mean(dim=1).squeeze()

# HuBERT 的单元量化
def hubert_quantization(hidden_states, codebook):
    """HuBERT 量化到离散单元"""
    # 计算与码本的距离
    distances = torch.cdist(hidden_states, codebook)

    # 选择最近的码本索引
    indices = torch.argmin(distances, dim=-1)

    return indices
```

### 3.1 语音识别系统

> [!example] 完整的语音识别流程

```python
import torch
import torchaudio
from transformers import Wav2Vec2Processor, Wav2Vec2ForCTC

class SpeechRecognitionSystem:
    def __init__(self, model_name="facebook/wav2vec2-base-960h"):
        """初始化语音识别系统"""
        self.processor = Wav2Vec2Processor.from_pretrained(model_name)
        self.model = Wav2Vec2ForCTC.from_pretrained(model_name)
        self.device = torch.device("cuda" if torch.cuda.is_available() else "cpu")
        self.model.to(self.device)

    def transcribe(self, audio_path):
        """语音转写"""
        # 加载音频
        audio, sr = torchaudio.load(audio_path)
        if sr != 16000:
            audio = torchaudio.transforms.Resample(sr, 16000)(audio)

        # 预处理
        input_values = self.processor(audio.squeeze(), return_tensors="pt", padding=True).input_values
        input_values = input_values.to(self.device)

        # 识别
        with torch.no_grad():
            logits = self.model(input_values).logits

        # 解码
        predicted_ids = torch.argmax(logits, dim=-1)
        transcription = self.processor.batch_decode(predicted_ids)

        return transcription[0]

    def batch_transcribe(self, audio_paths):
        """批量语音转写"""
        results = []
        for audio_path in audio_paths:
            result = self.transcribe(audio_path)
            results.append(result)
        return results

    def real_time_transcribe(self, audio_stream):
        """实时语音转写"""
        # 实时处理音频流
        input_values = self.processor(audio_stream, return_tensors="pt", padding=True).input_values
        input_values = input_values.to(self.device)

        with torch.no_grad():
            logits = self.model(input_values).logits

        predicted_ids = torch.argmax(logits, dim=-1)
        return self.processor.decode(predicted_ids[0])
```

### 3.2 说话人识别系统

> [!important] 说话人嵌入提取

```python
import torch
import torch.nn as nn
from sklearn.metrics.pairwise import cosine_similarity

class SpeakerRecognitionSystem:
    def __init__(self):
        """初始化说话人识别系统"""
        self.speaker_embeddings = {}
        self.embedding_size = 192  # x-vector 维度

    def extract_xvector(self, audio_path):
        """提取 x-vector 说话人嵌入"""
        # 加载音频
        audio, sr = torchaudio.load(audio_path)

        # 提取 MFCC 特征
        mfcc = librosa.feature.mfcc(y=audio.squeeze(), sr=sr, n_mfcc=20)

        # 简化的 x-vector 网络
        xvector_net = nn.Sequential(
            nn.Linear(mfcc.shape[0], 512),
            nn.ReLU(),
            nn.Linear(512, 256),
            nn.ReLU(),
            nn.Linear(256, self.embedding_size)
        )

        # 平均池化
        mfcc_mean = mfcc.mean(dim=1)
        embedding = xvector_net(mfcc_mean)

        return embedding.detach().numpy()

    def enroll_speaker(self, name, audio_files):
        """注册说话人"""
        embeddings = []
        for audio_file in audio_files:
            emb = self.extract_xvector(audio_file)
            embeddings.append(emb)

        # 平均多个样本
        self.speaker_embeddings[name] = np.mean(embeddings, axis=0)

    def recognize_speaker(self, audio_path, threshold=0.7):
        """识别说话人"""
        test_emb = self.extract_xvector(audio_path)

        similarities = {}
        for name, ref_emb in self.speaker_embeddings.items():
            sim = cosine_similarity([test_emb], [ref_emb])[0][0]
            similarities[name] = sim

        # 找到最相似的说话人
        best_match = max(similarities, key=similarities.get)
        max_sim = similarities[best_match]

        if max_sim > threshold:
            return best_match, max_sim
        else:
            return "unknown", max_sim
```

class AudioRetrievalSystem:
    def __init__(self, embedding_model="whisper"):
        """初始化音频检索系统"""
        self.embedding_model = embedding_model
        self.database = {}
        self.embeddings = []

    def add_audio(self, audio_id, audio_path, metadata=None):
        """添加音频到数据库"""
        # 提取嵌入
        if self.embedding_model == "whisper":
            embedding = extract_whisper_embedding(audio_path)
        else:
            embedding = extract_wav2vec_features(audio_path)

        self.database[audio_id] = {
            'path': audio_path,
            'embedding': embedding,
            'metadata': metadata or {}
        }
        self.embeddings.append(embedding)

    def search_similar(self, query_audio_path, top_k=5):
        """搜索相似音频"""
        # 提取查询嵌入
        if self.embedding_model == "whisper":
            query_emb = extract_whisper_embedding(query_audio_path)
        else:
            query_emb = extract_wav2vec_features(query_audio_path)

        # 计算相似度
        similarities = []
        for audio_id, data in self.database.items():
            sim = cosine_similarity([query_emb], [data['embedding']])[0][0]
            similarities.append((audio_id, sim))

        # 排序
        similarities.sort(key=lambda x: x[1], reverse=True)
        return similarities[:top_k]

    def semantic_search(self, text_query, top_k=5):
        """基于文本的语义搜索"""
        # 使用 Audio-CLIP 计算文本与音频的相似度
        results = []
        for audio_id, data in self.database.items():
            sim = audio_text_similarity(data['path'], text_query)
            results.append((audio_id, sim))

        # 排序
        results.sort(key=lambda x: x[1], reverse=True)
        return results[:top_k]

## 4. 最佳实践与优化策略

### 4.1 音频预处理流水线

> [!important] 标准化的音频预处理

```python
class AudioPreprocessor:
    def __init__(self, target_sr=16000, target_length=10.0):
        """初始化音频预处理器"""
        self.target_sr = target_sr
        self.target_length = target_length  # 秒
        self.target_length_samples = int(target_sr * target_length)

    def preprocess_audio(self, audio_path):
        """标准化音频预处理"""
        # 1. 加载音频
        audio, sr = torchaudio.load(audio_path)

        # 2. 重采样到目标采样率
        if sr != self.target_sr:
            resampler = torchaudio.transforms.Resample(sr, self.target_sr)
            audio = resampler(audio)

        # 3. 标准化长度
        if audio.shape[1] < self.target_length_samples:
            # 填充
            padding = self.target_length_samples - audio.shape[1]
            audio = torch.nn.functional.pad(audio, (0, padding))
        elif audio.shape[1] > self.target_length_samples:
            # 截断
            audio = audio[:, :self.target_length_samples]

        # 4. 归一化音频
        audio = (audio - audio.mean()) / (audio.std() + 1e-8)

        # 5. 音量标准化
        max_abs = torch.max(torch.abs(audio))
        if max_abs > 1.0:
            audio = audio / max_abs

        return audio

    def batch_preprocess(self, audio_paths):
        """批量预处理"""
        processed_audios = []
        for audio_path in audio_paths:
            processed = self.preprocess_audio(audio_path)
            processed_audios.append(processed)

        # 堆叠成一个批次
        return torch.stack(processed_audios)
```

class AudioDataset(Dataset):
    def __init__(self, audio_files, labels, preprocessor):
        self.audio_files = audio_files
        self.labels = labels
        self.preprocessor = preprocessor

    def __len__(self):
        return len(self.audio_files)

    def __getitem__(self, idx):
        audio = self.preprocessor.preprocess_audio(self.audio_files[idx])
        label = self.labels[idx]
        return audio, label

def finetune_audio_model(model, train_data, epochs=10, batch_size=32):
    """微调音频模型"""
    device = torch.device("cuda" if torch.cuda.is_available() else "cpu")
    model.to(device)

    # 数据加载
    dataset = AudioDataset(train_data['audio_files'], train_data['labels'], preprocessor)
    dataloader = DataLoader(dataset, batch_size=batch_size, shuffle=True)

    # 优化器
    optimizer = optim.Adam(model.parameters(), lr=1e-5)
    criterion = nn.CrossEntropyLoss()

    # 训练循环
    model.train()
    for epoch in range(epochs):
        total_loss = 0
        for batch in dataloader:
            audios, labels = batch
            audios = audios.to(device)
            labels = labels.to(device)

            # 前向传播
            outputs = model(audios)
            loss = criterion(outputs, labels)

            # 反向传播
            optimizer.zero_grad()
            loss.backward()
            optimizer.step()

            total_loss += loss.item()

        print(f"Epoch {epoch+1}, Loss: {total_loss/len(dataloader):.4f}")

class OptimizedAudioModel:
    def __init__(self, model_name):
        """初始化优化的音频模型"""
        self.model_name = model_name
        self.model = self.load_model()
        self.quantized = False
        self.cache = {}

    def load_model(self):
        """加载模型"""
        if "whisper" in self.model_name:
            processor = WhisperProcessor.from_pretrained(self.model_name)
            model = WhisperForConditionalGeneration.from_pretrained(self.model_name)
        else:
            processor = Wav2Vec2Processor.from_pretrained(self.model_name)
            model = Wav2Vec2Model.from_pretrained(self.model_name)

        # 量化模型
        model = torch.quantization.quantize_dynamic(
            model,
            {nn.Linear, nn.Conv1d},
            dtype=torch.qint8
        )

        return model

    def extract_embedding_cached(self, audio_path):
        """带缓存的嵌入提取"""
        # 检查缓存
        if audio_path in self.cache:
            return self.cache[audio_path]

        # 提取嵌入
        embedding = self.extract_embedding(audio_path)

        # 存入缓存
        self.cache[audio_path] = embedding

        return embedding

    def batch_extract_embeddings(self, audio_paths):
        """批量提取嵌入"""
        device = torch.device("cuda" if torch.cuda.is_available() else "cpu")
        embeddings = []

        for audio_path in audio_paths:
            emb = self.extract_embedding_cached(audio_path)
            embeddings.append(emb)

        return torch.stack(embeddings)
