# 评测体系：建立客观的评估标准

> [!abstract] 摘要


## 1. 评测维度

### 1.1 准确性维度  #LLM/评测

准确性是评估嵌入模型最核心的维度，主要衡量模型理解语义和匹配相关内容的能力。

> [!important] 评测任务类型
> 准确性评测主要涵盖以下四类任务：

1. **语义文本相似度** (STS)
   - 判断两个句子的语义相似度
   - 常用数据集：STS-B, STS-14
   - 评分范围：0-5 分

2. **检索排序**
   - 根据查询对文档进行相关性排序
   - 常用数据集：MSMARCO, TREC
   - 评估指标：NDCG@10, MRR

3. **分类任务**
   - 文本分类、情感分析、主题分类
   - 常用数据集：IMDB, 20 Newsgroups
   - 评估指标：Accuracy, F1-Score

4. **聚类任务**
   - 将相似文档分组
   - 常用数据集：Reuters-21578, 20 Newsgroups
   - 评估指标：NMI, ARI, Silhouette

```python
# 准确性评测实现
class AccuracyEvaluator:
    def __init__(self, model):
        self.model = model

    def evaluate_sts(self, sentence_pairs, gold_scores):
        """评估语义文本相似度"""
        predicted_scores = []

        for pair in sentence_pairs:
            sentence1, sentence2 = pair
            # 计算余弦相似度
            emb1 = self.model.encode(sentence1)
            emb2 = self.model.encode(sentence2)
            similarity = cosine_similarity(emb1.reshape(1, -1), emb2.reshape(1, -1))[0][0]
            predicted_scores.append(similarity)

        # 计算 Spearman 相关系数
        spearman = spearmanr(predicted_scores, gold_scores).correlation
        pearson = pearsonr(predicted_scores, gold_scores)[0]

        return {
            'spearman': spearman,
            'pearson': pearson,
            'mae': mean_absolute_error(predicted_scores, gold_scores)
        }

    def evaluate_retrieval(self, queries, documents, relevant_docs, top_k=10):
        """评估检索任务"""
        metrics = {
            'map': 0,
            'ndcg': 0,
            'mrr': 0,
            'recall': 0
        }
        num_queries = len(queries)

        for query in queries:
            # 执行检索
            query_emb = self.model.encode(query)
            doc_embs = self.model.encode_documents([doc['text'] for doc in documents])

            # 计算相似度
            similarities = cosine_similarity(query_emb.reshape(1, -1), doc_embs)[0]

            # 获取 top-k 结果
            top_indices = np.argsort(similarities)[::-1][:top_k]
            retrieved_docs = [documents[i]['id'] for i in top_indices]

            # 计算指标
            relevant_set = set(relevant_docs[query])
            retrieved_set = set(retrieved_docs)

            # MAP
            ap = self._calculate_ap(retrieved_docs, relevant_set)
            metrics['map'] += ap

            # NDCG
            ndcg = self._calculate_ndcg(similarities, relevant_set, top_k)
            metrics['ndcg'] += ndcg

            # MRR
            rr = self._calculate_rr(retrieved_docs, relevant_set)
            metrics['mrr'] += rr

            # Recall
            recall = len(relevant_set.intersection(retrieved_set)) / len(relevant_set)
            metrics['recall'] += recall

        # 计算平均值
        for metric in metrics:
            metrics[metric] /= num_queries

        return metrics

    def evaluate_clustering(self, documents, true_labels):
        """评估聚类任务"""
        # 获取文档嵌入
        doc_embs = self.model.encode_documents([doc['text'] for doc in documents])

        # K-means 聚类
        kmeans = KMeans(n_clusters=len(set(true_labels)))
        cluster_labels = kmeans.fit_predict(doc_embs)

        # 计算聚类指标
        nmi = normalized_mutual_info_score(true_labels, cluster_labels)
        ari = adjusted_rand_score(true_labels, cluster_labels)

        # Silhouette 分数
        silhouette = silhouette_score(doc_embs, cluster_labels)

        return {
            'nmi': nmi,
            'ari': ari,
            'silhouette': silhouette
        }

    def _calculate_ap(self, retrieved, relevant, k=10):
        """计算平均精度"""
        if len(retrieved) == 0:
            return 0

        retrieved = retrieved[:k]
        precision = 0
        relevant_count = 0

        for i, doc_id in enumerate(retrieved):
            if doc_id in relevant:
                relevant_count += 1
                precision += relevant_count / (i + 1)

        return precision / len(relevant) if len(relevant) > 0 else 0

    def _calculate_ndcg(self, similarities, relevant, k=10):
        """计算 NDCG"""
        top_k = np.argsort(similarities)[::-1][:k]
        dcg = 0
        idcg = 0

        for i, idx in enumerate(top_k):
            if idx in relevant:
                dcg += 1 / math.log2(i + 2)

        # 计算 IDCG
        ideal_relevance = sorted([1 if i in relevant else 0 for i in range(k)], reverse=True)
        for i, rel in enumerate(ideal_relevance):
            idcg += rel / math.log2(i + 2)

        return dcg / idcg if idcg > 0 else 0

    def _calculate_rr(self, retrieved, relevant):
        """计算倒数排名"""
        for i, doc_id in enumerate(retrieved):
            if doc_id in relevant:
                return 1.0 / (i + 1)
        return 0
```

### 1.2 效率维度  #LLM/检索技术

效率维度关注模型在实际应用中的性能表现，包括时间效率、空间效率等。

```python
# 效率评测实现
class EfficiencyEvaluator:
    def __init__(self, model):
        self.model = model

    def evaluate_latency(self, texts, batch_sizes=[1, 8, 32, 64]):
        """评估延迟性能"""
        results = []

        for batch_size in batch_sizes:
            # 预热
            _ = self.model.encode(["test"] * 10)

            # 测量延迟
            times = []
            for _ in range(100):
                start_time = time.time()
                _ = self.model.encode(texts[:batch_size])
                end_time = time.time()
                times.append(end_time - start_time)

            avg_time = np.mean(times)
            p95_time = np.percentile(times, 95)
            throughput = batch_size / avg_time

            results.append({
                'batch_size': batch_size,
                'avg_latency_ms': avg_time * 1000,
                'p95_latency_ms': p95_time * 1000,
                'throughput': throughput
            })

        return results

    def evaluate_memory_usage(self, texts):
        """评估内存使用"""
        # 测量编码前内存
        process = psutil.Process()
        memory_before = process.memory_info().rss / 1024 / 1024  # MB

        # 执行编码
        _ = self.model.encode(texts)

        # 测量编码后内存
        memory_after = process.memory_info().rss / 1024 / 1024  # MB

        memory_increase = memory_after - memory_before

        return {
            'memory_before_mb': memory_before,
            'memory_after_mb': memory_after,
            'memory_increase_mb': memory_increase,
            'memory_per_doc_mb': memory_increase / len(texts)
        }

    def evaluate_disk_space(self):
        """评估磁盘空间占用"""
        model_size = 0
        if hasattr(self.model, 'model'):
            # 检查模型大小
            for name, param in self.model.model.named_parameters():
                model_size += param.numel() * 4  # 假设 float32

            # 转换为 MB
            model_size_mb = model_size / 1024 / 1024
        else:
            model_size_mb = os.path.getsize(self.model.model_path) / 1024 / 1024

        return {
            'model_size_mb': model_size_mb,
            'embedding_size': 768 if hasattr(self.model, 'dim') else 1024
        }
```

### 1.3 鲁棒性维度

鲁棒性维度测试模型在各种干扰和异常情况下的表现。

```python
# 鲁棒性评测实现
class RobustnessEvaluator:
    def __init__(self, model):
        self.model = model

    def evaluate_noise_robustness(self, texts, noise_levels=[0.1, 0.2, 0.3]):
        """评估噪声鲁棒性"""
        results = []

        base_embeddings = self.model.encode(texts)

        for noise_level in noise_levels:
            noisy_embeddings = []
            for emb in base_embeddings:
                # 添加高斯噪声
                noise = np.random.normal(0, noise_level, emb.shape)
                noisy_emb = emb + noise
                noisy_embeddings.append(noisy_emb)

            # 计算相似度保持
            similarities = []
            for i, (clean, noisy) in enumerate(zip(base_embeddings, noisy_embeddings)):
                sim = cosine_similarity(clean.reshape(1, -1), noisy.reshape(1, -1))[0][0]
                similarities.append(sim)

            avg_similarity = np.mean(similarities)
            std_similarity = np.std(similarities)

            results.append({
                'noise_level': noise_level,
                'avg_similarity': avg_similarity,
                'std_similarity': std_similarity
            })

        return results

    def evaluate_perturbation_robustness(self, texts, perturbation_types=['synonym', 'delete', 'swap']):
        """评估扰动鲁棒性"""
        results = {}

        for perturbation in perturbation_types:
            perturbed_texts = self._apply_perturbation(texts, perturbation)
            perturbed_embeddings = self.model.encode(perturbed_texts)
            original_embeddings = self.model.encode(texts)

            # 计算相似度
            similarities = []
            for clean, pert in zip(original_embeddings, perturbed_embeddings):
                sim = cosine_similarity(clean.reshape(1, -1), pert.reshape(1, -1))[0][0]
                similarities.append(sim)

            results[perturbation] = {
                'avg_similarity': np.mean(similarities),
                'std_similarity': np.std(similarities),
                'min_similarity': np.min(similarities)
            }

        return results

    def evaluate_domain_shift(self, source_texts, target_texts):
        """评估领域迁移鲁棒性"""
        # 在源领域编码
        source_embeddings = self.model.encode(source_texts)
        target_embeddings = self.model.encode(target_texts)

        # 计算跨领域相似度
        cross_similarities = []
        for s_emb in source_embeddings[:10]:  # 取前 10 个样本
            for t_emb in target_embeddings[:10]:
                sim = cosine_similarity(s_emb.reshape(1, -1), t_emb.reshape(1, -1))[0][0]
                cross_similarities.append(sim)

        # 计算领域内相似度（基线）
        intra_source_sim = []
        for i in range(10):
            for j in range(i+1, 10):
                sim = cosine_similarity(
                    source_embeddings[i].reshape(1, -1),
                    source_embeddings[j].reshape(1, -1)
                )[0][0]
                intra_source_sim.append(sim)

        return {
            'cross_domain_similarity': np.mean(cross_similarities),
            'intra_domain_similarity': np.mean(intra_source_sim),
            'domain_gap': np.mean(intra_source_sim) - np.mean(cross_similarities)
        }

    def _apply_perturbation(self, texts, perturbation_type):
        """应用文本扰动"""
        perturbed = []

        for text in texts:
            if perturbation_type == 'synonym':
                # 同义词替换（简化版）
                words = text.split()
                # 这里应该调用真正的同义词词典
                perturbed_text = ' '.join(words)  # 简化实现
            elif perturbation_type == 'delete':
                # 删除随机词
                words = text.split()
                if len(words) > 1:
                    del_idx = np.random.randint(0, len(words))
                    words.pop(del_idx)
                perturbed_text = ' '.join(words)
            elif perturbation_type == 'swap':
                # 交换词序
                words = text.split()
                if len(words) > 1:
                    idx1, idx2 = np.random.choice(len(words), 2, replace=False)
                    words[idx1], words[idx2] = words[idx2], words[idx1]
                perturbed_text = ' '.join(words)
            else:
                perturbed_text = text

            perturbed.append(perturbed_text)

        return perturbed
```

### 2.1 MTEB 评测集  #LLM/评测

MTEB (Massive Text Embedding Benchmark) 是当前最全面的文本嵌入评测基准。

> [!summary] MTEB 概览
> MTEB 包含 59 个任务，覆盖 8 个语言，11 个评测任务类型。

```python
# MTEB 评测示例
class MTEBEvaluator:
    def __init__(self, model_name, tasks=None):
        self.model = SentenceTransformer(model_name)
        self.tasks = tasks or self._get_all_mteb_tasks()

    def _get_all_mteb_tasks(self):
        """获取所有 MTEB 任务"""
        # 简化实现，实际应该从 mteb 包加载
        return [
            'STS12', 'STS13', 'STS14', 'STS16',
            'SICK-R', 'MRPC', 'QQP', 'STS-B',
            'STS22', 'SummEval', 'ArguAna', 'FIQA',
            'HotpotQA', 'MSMARCO', 'NQ', 'Quora',
            'Touche2020', 'SciFact', 'FEVER', 'CQADupstack'
        ]

    def evaluate(self, split='test'):
        """执行 MTEB 评测"""
        results = {}

        for task_name in self.tasks:
            try:
                task = self._load_task(task_name)
                scores = self._evaluate_task(task, split)
                results[task_name] = scores
                print(f"{task_name}: {np.mean(list(scores.values())):.4f}")
            except Exception as e:
                print(f"Error evaluating {task_name}: {str(e)}")
                results[task_name] = {}

        # 计算总分
        avg_scores = []
        for scores in results.values():
            if scores:
                avg_scores.append(np.mean(list(scores.values())))

        results['average'] = np.mean(avg_scores)
        return results

    def _load_task(self, task_name):
        """加载评测任务"""
        # 这里应该从 mteb 包加载实际的任务
        # 简化实现
        if 'STS' in task_name:
            return {
                'type': 'sts',
                'data': self._load_sts_data(task_name)
            }
        elif 'QQP' in task_name:
            return {
                'type': 'classification',
                'data': self._load_qqp_data()
            }
        else:
            return {
                'type': 'retrieval',
                'data': self._load_retrieval_data(task_name)
            }

    def _evaluate_task(self, task, split):
        """评估单个任务"""
        if task['type'] == 'sts':
            return self._evaluate_sts(task['data'], split)
        elif task['type'] == 'classification':
            return self._evaluate_classification(task['data'], split)
        elif task['type'] == 'retrieval':
            return self._evaluate_retrieval(task['data'], split)

    def _load_sts_data(self, task_name):
        """加载 STS 数据"""
        # 简化实现
        return {
            'sentences1': ["sentence1", "sentence2"],
            'sentences2': ["sentence3", "sentence4"],
            'scores': [0.8, 0.5]
        }

    def _evaluate_sts(self, data, split):
        """评估 STS 任务"""
        predictions = []
        for s1, s2 in zip(data['sentences1'], data['sentences2']):
            emb1 = self.model.encode(s1)
            emb2 = self.model.encode(s2)
            sim = cosine_similarity(emb1.reshape(1, -1), emb2.reshape(1, -1))[0][0]
            predictions.append(sim)

        spearman = spearmanr(predictions, data['scores']).correlation
        return {'spearman': spearman}

    def generate_report(self, results):
        """生成评测报告"""
        # 按任务类型分组
        task_types = {
            'sts': [],
            'classification': [],
            'retrieval': [],
            'clustering': []
        }

        for task, score in results.items():
            if task != 'average':
                if 'STS' in task:
                    task_types['sts'].append(score)
                elif any(t in task for t in ['MRPC', 'QQP', 'SICK']):
                    task_types['classification'].append(score)
                elif task in ['T2Retrieval', 'MSMARCO']:
                    task_types['retrieval'].append(score)
                else:
                    task_types['clustering'].append(score)

        # 计算各类型平均分
        report = {}
        for task_type, scores in task_types.items():
            if scores:
                report[task_type] = {
                    'count': len(scores),
                    'average': np.mean(scores),
                    'std': np.std(scores)
                }

        # 总分
        report['overall'] = results['average']

        return report
```

```python
# BEIR 评测实现
class BEIREvaluator:
    def __init__(self, model_name):
        self.model = SentenceTransformer(model_name)
        self.dataset = 'beir'

    def evaluate_datasets(self, datasets=['msmarco', 'trec-covid', 'nfcorpus']):
        """评估多个 BEIR 数据集"""
        results = {}

        for dataset_name in datasets:
            try:
                scores = self._evaluate_dataset(dataset_name)
                results[dataset_name] = scores
                print(f"{dataset_name}: {scores['ndcg@10']:.4f}")
            except Exception as e:
                print(f"Error evaluating {dataset_name}: {str(e)}")

        return results

    def _evaluate_dataset(self, dataset_name):
        """评估单个数据集"""
        # 加载数据集
        corpus = self._load_corpus(dataset_name)
        queries = self._load_queries(dataset_name)
        qrels = self._load_qrels(dataset_name)

        # 编码文档和查询
        doc_embeddings = self.model.encode_documents(
            [doc['text'] for doc in corpus.values()],
            show_progress_bar=True
        )

        query_embeddings = {}
        for qid, query in queries.items():
            query_embeddings[qid] = self.model.encode(query)

        # 计算指标
        metrics = {}
        for qid, query_emb in query_embeddings.items():
            # 计算相似度
            similarities = cosine_similarity(
                query_emb.reshape(1, -1),
                doc_embeddings
            )[0]

            # 获取相关文档
            relevant_docs = set(qrels.get(qid, {}).keys())

            # 计算 NDCG
            top_k = 10
            ndcg = self._calculate_ndcg(similarities, relevant_docs, top_k)
            metrics.setdefault('ndcg@10', []).append(ndcg)

            # 计算 Recall
            recall = self._calculate_recall(similarities, relevant_docs, top_k)
            metrics.setdefault('recall@10', []).append(recall)

        # 计算平均
        for metric, values in metrics.items():
            metrics[metric] = np.mean(values)

        return metrics

    def _load_corpus(self, dataset_name):
        """加载文档语料"""
        # 简化实现
        return {
            'doc1': {'text': 'This is document 1'},
            'doc2': {'text': 'This is document 2'}
        }

    def _load_queries(self, dataset_name):
        """加载查询"""
        # 简化实现
        return {
            'q1': 'query 1',
            'q2': 'query 2'
        }

    def _load_qrels(self, dataset_name):
        """加载相关度判断"""
        # 简化实现
        return {
            'q1': {'doc1': 1, 'doc2': 0},
            'q2': {'doc1': 0, 'doc2': 1}
        }
```

## 3. 评测指标详解

### 3.1 准确率类指标

#### 平均精度 (MAP)

> [!important] MAP 计算公式
> $$\text{AP} = \frac{1}{R} \sum_{r=1}^{R} P(r) \cdot \text{rel}(r)$$
> $$\text{MAP} = \frac{1}{Q} \sum_{q=1}^{Q} \text{AP}(q)$$

其中 $R$ 是相关文档数，$Q$ 是查询数，$\text{rel}(r)$ 是指示函数。

```python
def calculate_map(retrieved_lists, relevant_lists, k=10):
    """计算 MAP"""
    ap_scores = []

    for retrieved, relevant in zip(retrieved_lists, relevant_lists):
        relevant = set(relevant)
        retrieved = retrieved[:k]

        precision_sum = 0
        relevant_count = 0

        for i, doc_id in enumerate(retrieved):
            if doc_id in relevant:
                relevant_count += 1
                precision_sum += relevant_count / (i + 1)

        ap = precision_sum / len(relevant) if len(relevant) > 0 else 0
        ap_scores.append(ap)

    return np.mean(ap_scores)
```

#### 归一化折损累计增益 (NDCG)

NDCG 考虑了排序位置，位置越靠前的相关文档贡献越大。

```python
def calculate_ndcg(retrieved_ids, relevant_dict, k=10):
    """计算 NDCG"""
    # DCG
    dcg = 0
    for i, doc_id in enumerate(retrieved_ids[:k]):
        relevance = relevant_dict.get(doc_id, 0)
        dcg += relevance / math.log2(i + 2)

    # IDCG
    ideal_relevance = sorted(relevant_dict.values(), reverse=True)[:k]
    idcg = 0
    for i, rel in enumerate(ideal_relevance):
        idcg += rel / math.log2(i + 2)

    return dcg / idcg if idcg > 0 else 0
```

### 3.2 排序类指标

#### 倒数排名 (MRR)

MRR 关注第一个相关文档的位置。

```python
def calculate_mrr(retrieved_lists, relevant_lists):
    """计算 MRR"""
    rr_scores = []

    for retrieved, relevant in zip(retrieved_lists, relevant_lists):
        relevant = set(relevant)
        rr = 0

        for i, doc_id in enumerate(retrieved):
            if doc_id in relevant:
                rr = 1.0 / (i + 1)
                break

        rr_scores.append(rr)

    return np.mean(rr_scores)
```

#### 精确率与召回率

```python
def calculate_precision_recall(retrieved, relevant, k=10):
    """计算精确率和召回率"""
    retrieved = set(retrieved[:k])
    relevant = set(relevant)

    true_positives = len(retrieved.intersection(relevant))
    false_positives = len(retrieved - relevant)
    false_negatives = len(relevant - retrieved)

    precision = true_positives / (true_positives + false_positives) if (true_positives + false_positives) > 0 else 0
    recall = true_positives / (true_positives + false_negatives) if (true_positives + false_negatives) > 0 else 0

    return precision, recall
```

### 3.3 多样性指标

#### 排档多样性

测量返回结果的主题多样性。

```python
def calculate_diversity(retrieved_docs, topic_model, k=10):
    """计算排档多样性"""
    # 获取文档主题分布
    doc_topics = topic_model.transform(retrieved_docs[:k])

    # 计算平均主题分布
    avg_topic_dist = np.mean(doc_topics, axis=0)

    # 计算熵作为多样性指标
    entropy = -np.sum(avg_topic_dist * np.log(avg_topic_dist + 1e-10))

    return entropy
```

#### 覆盖率

测量返回结果覆盖了多少相关文档。

```python
def calculate_coverage(retrieved_lists, all_relevant):
    """计算覆盖率"""
    total_relevant = set()
    for rel_list in all_relevant:
        total_relevant.update(rel_list)

    retrieved_union = set()
    for ret_list in retrieved_lists:
        retrieved_union.update(ret_list)

    coverage = len(retrieved_union.intersection(total_relevant)) / len(total_relevant)

    return coverage
```

```python
def calculate_throughput(latency_results):
    """计算吞吐量"""
    # 使用平均延迟计算吞吐量
    avg_latency = np.mean([r['avg_latency_ms'] for r in latency_results])
    throughput_per_sec = 1000 / avg_latency  # 每秒处理请求数

    # 考虑批处理
    batch_throughputs = []
    for result in latency_results:
        batch_size = result['batch_size']
        batch_latency = result['avg_latency_ms']
        batch_throughput = (batch_size * 1000) / batch_latency
        batch_throughputs.append(batch_throughput)

    return {
        'throughput_per_request': throughput_per_sec,
        'batch_throughput': np.mean(batch_throughputs)
    }
```

```python
# 评测执行器
class BenchmarkRunner:
    def __init__(self, config):
        self.config = config
        self.results = {}

    def run_benchmark(self, model_name, save_results=True):
        """运行完整评测"""
        model = SentenceTransformer(self.config.models[model_name])
        results = {}

        # 准确性评测
        print(f"Running accuracy evaluation for {model_name}...")
        accuracy_evaluator = AccuracyEvaluator(model)
        accuracy_results = accuracy_evaluator.evaluate_mteb()
        results['accuracy'] = accuracy_results

        # 效率评测
        print(f"Running efficiency evaluation for {model_name}...")
        efficiency_evaluator = EfficiencyEvaluator(model)
        efficiency_results = efficiency_evaluator.evaluate_latency(
            sample_texts, self.config.benchmark_settings['batch_sizes']
        )
        results['efficiency'] = efficiency_results

        # 鲁棒性评测
        print(f"Running robustness evaluation for {model_name}...")
        robustness_evaluator = RobustnessEvaluator(model)
        robustness_results = robustness_evaluator.evaluate_noise_robustness(
            sample_texts
        )
        results['robustness'] = robustness_results

        # 保存结果
        if save_results:
            self.results[model_name] = results
            self._save_results(model_name, results)

        return results

    def compare_models(self, model_names=None):
        """比较多个模型"""
        if model_names is None:
            model_names = list(self.config.models.keys())

        comparison = {}
        for model_name in model_names:
            if model_name not in self.results:
                self.run_benchmark(model_name)

            comparison[model_name] = self._format_results(self.results[model_name])

        return comparison

    def _save_results(self, model_name, results):
        """保存评测结果"""
        timestamp = datetime.now().strftime("%Y%m%d_%H%M%S")
        filename = f"benchmark_results_{model_name}_{timestamp}.json"

        with open(filename, 'w') as f:
            json.dump(results, f, indent=2, cls=NumpyEncoder)

    def _format_results(self, results):
        """格式化结果以便比较"""
        formatted = {}

        # 准确性指标
        if 'accuracy' in results:
            formatted['accuracy'] = {
                'mteb_score': results['accuracy'].get('average', 0),
                'sts_avg': np.mean([
                    results['accuracy'].get(task, {}).get('spearman', 0)
                    for task in self.config.datasets['mteb'] if 'STS' in task
                ])
            }

        # 效率指标
        if 'efficiency' in results:
            latency_results = results['efficiency']
            formatted['efficiency'] = {
                'avg_latency_ms': np.mean([
                    r['avg_latency_ms'] for r in latency_results
                ]),
                'throughput': np.mean([
                    r['throughput'] for r in latency_results
                ])
            }

        # 鲁棒性指标
        if 'robustness' in results:
            formatted['robustness'] = {
                'noise_robustness': np.mean([
                    r['avg_similarity'] for r in results['robustness']
                ])
            }

        return formatted
```

```python
# 标准化评测流程
class StandardizedEvaluationPipeline:
    def __init__(self, config):
        self.config = config
        self.preprocessor = TextPreprocessor()
        self.setup_environment()

    def setup_environment(self):
        """设置评测环境"""
        # 设置随机种子
        np.random.seed(42)
        torch.manual_seed(42)

        # 检查硬件
        if torch.cuda.is_available():
            self.device = 'cuda'
            torch.backends.cudnn.deterministic = True
        else:
            self.device = 'cpu'

        print(f"Using device: {self.device}")

    def evaluate_model(self, model_name):
        """标准化的模型评测流程"""
        print(f"\n=== Evaluating {model_name} ===")

        # 1. 加载模型
        model = self.load_model(model_name)

        # 2. 预处理数据
        datasets = self.load_datasets()

        # 3. 运行评测
        results = {}
        for dataset_name, dataset in datasets.items():
            print(f"\nEvaluating on {dataset_name}...")
            results[dataset_name] = self.evaluate_on_dataset(model, dataset)

        # 4. 计算综合得分
        overall_score = self.calculate_overall_score(results)
        results['overall'] = overall_score

        # 5. 生成报告
        self.generate_report(model_name, results)

        return results

    def load_model(self, model_name):
        """加载模型"""
        model_path = self.config.models.get(model_name)
        if model_path:
            model = SentenceTransformer(model_path)
            model.to(self.device)
            return model
        else:
            raise ValueError(f"Model {model_name} not found in config")

    def load_datasets(self):
        """加载评测数据集"""
        datasets = {}

        # 加载 MTEB 数据集
        for task in self.config.datasets['mteb']:
            datasets[task] = load_mteb_task(task)

        # 加载 BEIR 数据集
        for dataset in self.config.datasets['beir']:
            datasets[dataset] = load_beir_dataset(dataset)

        # 加载中文数据集
        for dataset in self.config.datasets['chinese']:
            datasets[dataset] = load_chinese_dataset(dataset)

        return datasets

    def generate_report(self, model_name, results):
        """生成评测报告"""
        timestamp = datetime.now().strftime("%Y%m%d_%H%M%S")

        # 生成详细报告
        report = {
            'model_name': model_name,
            'evaluation_time': timestamp,
            'device': self.device,
            'results': results,
            'summary': {
                'overall_score': results['overall'],
                'best_dataset': max(results.items(), key=lambda x: x[1]['score'])[0],
                'worst_dataset': min(results.items(), key=lambda x: x[1]['score'])[0]
            }
        }

        # 保存报告
        filename = f"evaluation_report_{model_name}_{timestamp}.json"
        with open(filename, 'w') as f:
            json.dump(report, f, indent=2, cls=NumpyEncoder)

        print(f"\nReport saved to: {filename}")
        return report
```
